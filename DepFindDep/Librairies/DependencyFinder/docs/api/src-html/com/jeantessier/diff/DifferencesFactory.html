033    package com.jeantessier.diff;
034    
035    import java.util.*;
036    
037    import org.apache.log4j.*;
038    
039    import com.jeantessier.classreader.*;
040    
041    public class DifferencesFactory {
042        private Classfile oldClass;
043        private Classfile newClass;
044    
045        private DifferenceStrategy strategy;
046    
047        /**
048         * For tests only.
049         */
050        DifferencesFactory() {
051            this(new APIDifferenceStrategy(new CodeDifferenceStrategy()));
052        }
053    
054        public DifferencesFactory(DifferenceStrategy strategy) {
055            this.strategy = strategy;
056        }
057    
058        public Differences createProjectDifferences(String name, String oldVersion, PackageMapper oldPackages, String newVersion, PackageMapper newPackages) {
059            Logger.getLogger(getClass()).debug("Begin " + name + " (" + oldVersion + " -> " + newVersion + ")");
060    
061            ProjectDifferences projectDifferences = new ProjectDifferences(name, oldVersion, newVersion);
062    
063            Logger.getLogger(getClass()).debug("      Collecting packages ...");
064    
065            Collection<String> packageNames = new TreeSet<String>();
066            packageNames.addAll(oldPackages.getPackageNames());
067            packageNames.addAll(newPackages.getPackageNames());
068    
069            Logger.getLogger(getClass()).debug("      Diff'ing packages ...");
070    
071            for (String packageName : packageNames) {
072                Map<String, Classfile> oldPackage = oldPackages.getPackage(packageName);
073                if (oldPackage == null) {
074                    oldPackage = Collections.EMPTY_MAP;
075                }
076    
077                Map<String, Classfile> newPackage = newPackages.getPackage(packageName);
078                if (newPackage == null) {
079                    newPackage = Collections.EMPTY_MAP;
080                }
081    
082                if (strategy.isPackageDifferent(oldPackage, newPackage)) {
083                    projectDifferences.getPackageDifferences().add(createPackageDifferences(packageName, oldPackage, newPackage));
084                }
085            }
086    
087            Logger.getLogger(getClass()).debug("End   " + name + " (" + oldVersion + " -> " + newVersion + ")");
088    
089            return projectDifferences;
090        }
091    
092        public Differences createPackageDifferences(String name, Map<String, Classfile> oldPackage, Map<String, Classfile> newPackage) {
093            Logger.getLogger(getClass()).debug("Begin " + name);
094    
095            PackageDifferences packageDifferences = new PackageDifferences(name, oldPackage, newPackage);
096    
097            if (oldPackage != null && !oldPackage.isEmpty() && newPackage != null && !newPackage.isEmpty()) {
098                Logger.getLogger(getClass()).debug("      Diff'ing classes ...");
099    
100                Collection<String> classNames = new TreeSet<String>();
101                classNames.addAll(oldPackage.keySet());
102                classNames.addAll(newPackage.keySet());
103    
104                for (String className : classNames) {
105                    Classfile oldClass = oldPackage.get(className);
106                    Classfile newClass = newPackage.get(className);
107    
108                    if (strategy.isClassDifferent(oldClass, newClass)) {
109                        packageDifferences.getClassDifferences().add(createClassDifferences(className, oldClass, newClass));
110                    }
111                }
112    
113                Logger.getLogger(getClass()).debug("      " + name + " has " + packageDifferences.getClassDifferences().size() + " class(es) that changed.");
114            }
115    
116            Logger.getLogger(getClass()).debug("End   " + name);
117    
118            return packageDifferences;
119        }
120    
121        public Differences createClassDifferences(String name, Classfile oldClass, Classfile newClass) {
122            Logger.getLogger(getClass()).debug("Begin " + name);
123    
124            ClassDifferences classDifferences;
125            if (((oldClass != null) && oldClass.isInterface()) || ((newClass != null) && newClass.isInterface())) {
126                classDifferences = new InterfaceDifferences(name, oldClass, newClass);
127            } else {
128                classDifferences = new ClassDifferences(name, oldClass, newClass);
129            }
130    
131            if (!classDifferences.isRemoved() && !classDifferences.isNew() && strategy.isDeclarationModified(oldClass, newClass)) {
132                classDifferences.setDeclarationModified(true);
133            }
134    
135            Differences result = classDifferences;
136    
137            this.oldClass = oldClass;
138            this.newClass = newClass;
139    
140            if (oldClass != null && newClass != null) {
141                Logger.getLogger(getClass()).debug("      Collecting fields ...");
142    
143                Map<String, String> fieldLevel = new TreeMap<String, String>();
144    
145                for (Field_info field : oldClass.getAllFields()) {
146                    fieldLevel.put(field.getName(), field.getFullSignature());
147                }
148    
149                for (Field_info field : newClass.getAllFields()) {
150                    fieldLevel.put(field.getName(), field.getFullSignature());
151                }
152    
153                Logger.getLogger(getClass()).debug("      Diff'ing fields ...");
154    
155                for (String fieldName : fieldLevel.keySet()) {
156                    String fieldFullName = fieldLevel.get(fieldName);
157    
158                    Field_info oldField = oldClass.getField(fieldName);
159                    Field_info newField = newClass.getField(fieldName);
160    
161                    if (strategy.isFieldDifferent(oldField, newField)) {
162                        classDifferences.getFeatureDifferences().add(createFeatureDifferences(fieldFullName, oldField, newField));
163                    }
164                }
165    
166                Logger.getLogger(getClass()).debug("      Collecting methods ...");
167    
168                Map<String, String> methodLevel = new TreeMap<String, String>();
169    
170                for (Method_info method : oldClass.getAllMethods()) {
171                    methodLevel.put(method.getSignature(), method.getFullSignature());
172                }
173    
174                for (Method_info method : newClass.getAllMethods()) {
175                    methodLevel.put(method.getSignature(), method.getFullSignature());
176                }
177    
178                Logger.getLogger(getClass()).debug("      Diff'ing methods ...");
179    
180                for (String methodName : methodLevel.keySet()) {
181                    String methodFullName = methodLevel.get(methodName);
182    
183                    Method_info oldMethod = oldClass.getMethod(methodName);
184                    Method_info newMethod = newClass.getMethod(methodName);
185    
186                    if (strategy.isMethodDifferent(oldMethod, newMethod)) {
187                        classDifferences.getFeatureDifferences().add(createFeatureDifferences(methodFullName, oldMethod, newMethod));
188                    }
189                }
190    
191                Logger.getLogger(getClass()).debug(name + " has " + classDifferences.getFeatureDifferences().size() + " feature(s) that changed.");
192    
193                if (oldClass.isDeprecated() != newClass.isDeprecated()) {
194                    result = new DeprecatableDifferences(result, oldClass, newClass);
195                }
196            }
197    
198            Logger.getLogger(getClass()).debug("End   " + name);
199    
200            return result;
201        }
202    
203        public Differences createFeatureDifferences(String name, Feature_info oldFeature, Feature_info newFeature) {
204            Logger.getLogger(getClass()).debug("Begin " + name);
205    
206            FeatureDifferences featureDifferences;
207            if (oldFeature instanceof Field_info || newFeature instanceof Field_info) {
208                featureDifferences = new FieldDifferences(name, (Field_info) oldFeature, (Field_info) newFeature);
209    
210                if (!featureDifferences.isRemoved() && !featureDifferences.isNew() && strategy.isConstantValueDifferent(((Field_info) oldFeature).getConstantValue(), ((Field_info) newFeature).getConstantValue())) {
211                    ((FieldDifferences) featureDifferences).setConstantValueDifference(true);
212                }
213    
214                if (featureDifferences.isRemoved() && newClass.locateField(name) != null) {
215                    featureDifferences.setInherited(true);
216                }
217            } else {
218                if (((oldFeature instanceof Method_info) && ((Method_info) oldFeature).isConstructor()) || ((newFeature instanceof Method_info) && ((Method_info) newFeature).isConstructor())) {
219                    featureDifferences = new ConstructorDifferences(name, (Method_info) oldFeature, (Method_info) newFeature);
220                } else {
221                    featureDifferences = new MethodDifferences(name, (Method_info) oldFeature, (Method_info) newFeature);
222                }
223    
224                if (!featureDifferences.isRemoved() && !featureDifferences.isNew() && strategy.isCodeDifferent(((Method_info) oldFeature).getCode(), ((Method_info) newFeature).getCode())) {
225                    ((CodeDifferences) featureDifferences).setCodeDifference(true);
226                }
227    
228                if (featureDifferences.isRemoved()) {
229                    Method_info attempt = newClass.locateMethod(name);
230                    if ((attempt != null) && (oldFeature.getClassfile().isInterface() == attempt.getClassfile().isInterface())) {
231                        featureDifferences.setInherited(true);
232                    }
233                }
234            }
235    
236            Differences result = featureDifferences;
237            
238            if (oldFeature != null && newFeature != null) {
239                if (oldFeature.isDeprecated() != newFeature.isDeprecated()) {
240                    result = new DeprecatableDifferences(result, oldFeature, newFeature);
241                }
242            }
243    
244            Logger.getLogger(getClass()).debug("End   " + name);
245    
246            return result;
247        }
