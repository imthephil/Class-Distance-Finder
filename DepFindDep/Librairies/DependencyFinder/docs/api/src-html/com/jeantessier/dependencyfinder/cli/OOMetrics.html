051        protected void showSpecificUsage(PrintStream out) {
052            out.println();
053            out.println("If no files are specified, it processes the current directory.");

056            out.println("ending in \".class\".");
057            out.println();
058            out.println("Defaults is text output to the console.");
059            out.println();
060        }

062        protected void populateCommandLineSwitches() {
063            super.populateCommandLineSwitches();
064            populateCommandLineSwitchesForXMLOutput(com.jeantessier.metrics.XMLPrinter.DEFAULT_ENCODING, com.jeantessier.metrics.XMLPrinter.DEFAULT_DTD_PREFIX);
065    
066            getCommandLine().addSingleValueSwitch("project-name",           DEFAULT_PROJECT_NAME);
067            getCommandLine().addSingleValueSwitch("default-configuration", true);
068            getCommandLine().addSingleValueSwitch("configuration");
069            getCommandLine().addToggleSwitch("csv");
070            getCommandLine().addToggleSwitch("txt");
071            getCommandLine().addToggleSwitch("xml");
072            getCommandLine().addToggleSwitch("validate");
073            getCommandLine().addToggleSwitch("project");
074            getCommandLine().addToggleSwitch("groups");
075            getCommandLine().addToggleSwitch("classes");
076            getCommandLine().addToggleSwitch("methods");

078            getCommandLine().addMultipleValuesSwitch("scope-excludes-list");
079            getCommandLine().addMultipleValuesSwitch("filter-includes-list");
080            getCommandLine().addMultipleValuesSwitch("filter-excludes-list");
081            getCommandLine().addToggleSwitch("show-all-metrics");
082            getCommandLine().addToggleSwitch("show-empty-metrics");
083            getCommandLine().addToggleSwitch("show-hidden-measurements");
084            getCommandLine().addSingleValueSwitch("sort",                   DEFAULT_SORT);
085            getCommandLine().addToggleSwitch("expand");
086            getCommandLine().addToggleSwitch("reverse");
087        }

089        protected Collection<CommandLineException> parseCommandLine(String[] args) {
090            Collection<CommandLineException> exceptions = super.parseCommandLine(args);
091    
092            if (!getCommandLine().isPresent("project") && !getCommandLine().isPresent("groups") && !getCommandLine().isPresent("classes") && !getCommandLine().isPresent("methods")) {
093                getCommandLine().getSwitch("project").setValue(true);
094                getCommandLine().getSwitch("groups").setValue(true);
095                getCommandLine().getSwitch("classes").setValue(true);
096                getCommandLine().getSwitch("methods").setValue(true);
097            }
098    
099            int modeSwitch = 0;
100    
101            if (getCommandLine().getToggleSwitch("csv")) {
102                modeSwitch++;
103            }
104            if (getCommandLine().getToggleSwitch("txt")) {
105                modeSwitch++;
106            }
107            if (getCommandLine().getToggleSwitch("xml")) {
108                modeSwitch++;
109            }
110            if (modeSwitch != 1) {
111                exceptions.add(new CommandLineException("Must have one and only one of -csv, -txt, or -xml"));
112            }
113    
114            return exceptions;
115        }

117        protected void doProcessing() throws Exception {
118            Logger.getLogger(OOMetrics.class).debug("Reading sources ...");
119            getVerboseListener().print("Reading sources ...");
120    
121            List<String> parameters = getCommandLine().getParameters();
122            if (parameters.size() == 0) {
123                parameters.add(".");
124            }
125    
126            ClassfileLoader loader = new AggregatingClassfileLoader();
127            loader.addLoadListener(getVerboseListener());
128            loader.load(parameters);
129    
130            Logger.getLogger(OOMetrics.class).debug("Reading configuration ...");
131            getVerboseListener().print("Reading configuration ...");
132    
133            String projectName = getCommandLine().getSingleSwitch("project-name");
134    
135            MetricsFactory factory;
136            if (getCommandLine().isPresent("configuration")) {
137                factory = new MetricsFactory(projectName, new MetricsConfigurationLoader(getCommandLine().getToggleSwitch("validate")).load(getCommandLine().getSingleSwitch("configuration")));
138            } else {
139                factory = new MetricsFactory(projectName, new MetricsConfigurationLoader(getCommandLine().getToggleSwitch("validate")).load(getCommandLine().getSingleSwitch("default-configuration")));
140            }
141    
142            Logger.getLogger(OOMetrics.class).debug("Computing metrics ...");
143            getVerboseListener().print("Computing metrics ...");
144    
145            com.jeantessier.metrics.MetricsGatherer gatherer = new com.jeantessier.metrics.MetricsGatherer(projectName, factory);
146            if (getCommandLine().isPresent("scope-includes-list") || getCommandLine().isPresent("scope-excludes-list")) {
147                gatherer.setScopeIncludes(createCollection(getCommandLine().getMultipleSwitch("scope-includes-list"), getCommandLine().getMultipleSwitch("scope-excludes-list")));
148            }
149            if (getCommandLine().isPresent("filter-includes-list") || getCommandLine().isPresent("filter-excludes-list")) {
150                gatherer.setFilterIncludes(createCollection(getCommandLine().getMultipleSwitch("filter-includes-list"), getCommandLine().getMultipleSwitch("filter-excludes-list")));
151            }
152            gatherer.addMetricsListener(getVerboseListener());
153            gatherer.visitClassfiles(loader.getAllClassfiles());
154    
155            if (getCommandLine().isPresent("show-all-metrics")) {
156                Iterator i;
157    
158                i = gatherer.getMetricsFactory().getAllClassMetrics().iterator();
159                while (i.hasNext()) {
160                    gatherer.getMetricsFactory().includeClassMetrics((Metrics) i.next());
161                }
162    
163                i = gatherer.getMetricsFactory().getAllMethodMetrics().iterator();
164                while (i.hasNext()) {
165                    gatherer.getMetricsFactory().includeMethodMetrics((Metrics) i.next());
166                }
167            }
168    
169            Logger.getLogger(OOMetrics.class).debug("Printing results ...");
170            getVerboseListener().print("Printing results ...");
171    
172            if (getCommandLine().isPresent("csv")) {
173                printCSVFiles(gatherer.getMetricsFactory());
174            } else if (getCommandLine().isPresent("txt")) {
175                printTextFile(gatherer.getMetricsFactory());
176            } else if (getCommandLine().isPresent("xml")) {
177                printXMLFile(gatherer.getMetricsFactory());
178            }
179    
180            Logger.getLogger(OOMetrics.class).debug("Done.");
181        }

183        private static Collection<String> createCollection(Collection<String> includes, Collection<String> excludes) throws IOException {
184            Collection<String> result = new HashSet<String>();
185    
186            for (String include : includes) {
187                BufferedReader reader = new BufferedReader(new FileReader(include));
188                String line;
189                while ((line = reader.readLine()) != null) {
190                    result.add(line);
191                }
192                reader.close();
193            }
194    
195            for (String exclude : excludes) {
196                BufferedReader reader = new BufferedReader(new FileReader(exclude));
197                String line;
198                while ((line = reader.readLine()) != null) {
199                    result.remove(line);
200                }
201                reader.close();
202            }
203    
204            return result;
205        }

207        private void printCSVFiles(MetricsFactory factory) throws IOException {
208            MetricsComparator comparator = new MetricsComparator(getCommandLine().getSingleSwitch("sort"));
209            if (getCommandLine().getToggleSwitch("reverse")) {
210                comparator.reverse();
211            }
212    
213            List<Metrics> metrics;
214            com.jeantessier.metrics.Printer printer;
215    
216            if (getCommandLine().getToggleSwitch("project")) {
217                if (getCommandLine().isPresent("out")) {
218                    out = new PrintWriter(new FileWriter(getCommandLine().getSingleSwitch("out") + "_project.csv"));
219                } else {
220                    out.println("Project:");
221                }
222    
223                metrics = new ArrayList<Metrics>(factory.getProjectMetrics());
224                Collections.sort(metrics, comparator);
225                printer = new com.jeantessier.metrics.CSVPrinter(out, factory.getConfiguration().getProjectMeasurements());
226                printer.setShowEmptyMetrics(getCommandLine().isPresent("show-empty-metrics"));
227                printer.setShowHiddenMeasurements(getCommandLine().isPresent("show-hidden-measurements"));
228                if (getCommandLine().isPresent("indent-text")) {
229                    printer.setIndentText(getCommandLine().getSingleSwitch("indent-text"));
230                }
231    
232                printer.visitMetrics(metrics);
233    
234                if (getCommandLine().isPresent("out")) {
235                    out.close();
236                } else {
237                    out.println();
238                }
239            }
240    
241            if (getCommandLine().getToggleSwitch("groups")) {
242                if (getCommandLine().isPresent("out")) {
243                    out = new PrintWriter(new FileWriter(getCommandLine().getSingleSwitch("out") + "_groups.csv"));
244                } else {
245                    out.println("Packages:");
246                }
247    
248                metrics = new ArrayList<Metrics>(factory.getGroupMetrics());
249                Collections.sort(metrics, comparator);
250                printer = new com.jeantessier.metrics.CSVPrinter(out, factory.getConfiguration().getGroupMeasurements());
251                printer.setShowEmptyMetrics(getCommandLine().isPresent("show-empty-metrics"));
252                printer.setShowHiddenMeasurements(getCommandLine().isPresent("show-hidden-measurements"));
253                if (getCommandLine().isPresent("indent-text")) {
254                    printer.setIndentText(getCommandLine().getSingleSwitch("indent-text"));
255                }
256    
257                printer.visitMetrics(metrics);
258    
259                if (getCommandLine().isPresent("out")) {
260                    out.close();
261                } else {
262                    out.println();
263                }
264            }
265    
266            if (getCommandLine().getToggleSwitch("classes")) {
267                if (getCommandLine().isPresent("out")) {
268                    out = new PrintWriter(new FileWriter(getCommandLine().getSingleSwitch("out") + "_classes.csv"));
269                } else {
270                    out.println("Classes:");
271                }
272    
273                metrics = new ArrayList<Metrics>(factory.getClassMetrics());
274                Collections.sort(metrics, comparator);
275                printer = new com.jeantessier.metrics.CSVPrinter(out, factory.getConfiguration().getClassMeasurements());
276                printer.setShowEmptyMetrics(getCommandLine().isPresent("show-empty-metrics"));
277                printer.setShowHiddenMeasurements(getCommandLine().isPresent("show-hidden-measurements"));
278                if (getCommandLine().isPresent("indent-text")) {
279                    printer.setIndentText(getCommandLine().getSingleSwitch("indent-text"));
280                }
281    
282                printer.visitMetrics(metrics);
283    
284                if (getCommandLine().isPresent("out")) {
285                    out.close();
286                } else {
287                    out.println();
288                }
289            }
290    
291            if (getCommandLine().getToggleSwitch("methods")) {
292                if (getCommandLine().isPresent("out")) {
293                    out = new PrintWriter(new FileWriter(getCommandLine().getSingleSwitch("out") + "_methods.csv"));
294                } else {
295                    out.println("Methods:");
296                }
297    
298                metrics = new ArrayList<Metrics>(factory.getMethodMetrics());
299                Collections.sort(metrics, comparator);
300                printer = new com.jeantessier.metrics.CSVPrinter(out, factory.getConfiguration().getMethodMeasurements());
301                printer.setShowEmptyMetrics(getCommandLine().isPresent("show-empty-metrics"));
302                printer.setShowHiddenMeasurements(getCommandLine().isPresent("show-hidden-measurements"));
303                if (getCommandLine().isPresent("indent-text")) {
304                    printer.setIndentText(getCommandLine().getSingleSwitch("indent-text"));
305                }
306    
307                printer.visitMetrics(metrics);
308    
309                if (getCommandLine().isPresent("out")) {
310                    out.close();
311                }
312            }
