599        public String getMnemonic() {
600            String result = getMnemonic(getOpcode());
601    
602            if (getOpcode() == 0xc4 /* wide */) {
603                result += " " + getMnemonic(bytecode[start+1] & 0xff);
604            }
605    
606            return result;
607        }

609        public int getLength() {
610            int result = length[getOpcode()];
611    
612            int padding, low, high, npairs;
613        
614            switch (getOpcode()) {
615                case 0xaa:
616                    // tableswitch
617                    padding = 3 - (start % 4);
618                    low =
619                        ((bytecode[start+padding+5] & 0xff) << 24) |
620                        ((bytecode[start+padding+6] & 0xff) << 16) |
621                        ((bytecode[start+padding+7] & 0xff) << 8)  |
622                        ((bytecode[start+padding+8] & 0xff));
623                    high =
624                        ((bytecode[start+padding+9] & 0xff)  << 24) |
625                        ((bytecode[start+padding+10] & 0xff) << 16) |
626                        ((bytecode[start+padding+11] & 0xff) << 8)  |
627                        ((bytecode[start+padding+12] & 0xff));
628                    result =
629                        1 +                   // opcode
630                        padding +             // padding
631                        12 +                  // default + low + high
632                        (high - low + 1) * 4; // (high - low + 1) * offset
633                    break;
634    
635                case 0xab:
636                    // lookupswitch
637                    padding = 3 - (start % 4);
638                    npairs =
639                        ((bytecode[start+padding+5] & 0xff) << 24) |
640                        ((bytecode[start+padding+6] & 0xff) << 16) |
641                        ((bytecode[start+padding+7] & 0xff) << 8)  |
642                        ((bytecode[start+padding+8] & 0xff));
643                    result =
644                        1 +            // opcode
645                        padding +      // padding
646                        8 +            // default + npairs
647                        (npairs * 8);  // npairs * (match + offset)
648                    break;
649    
650                case 0xc4:
651                    // wide
652                    if ((bytecode[start+1] & 0xff) == 0x84 /* iinc */) {
653                        result = 6;
654                    } else {
655                        result = 4;
656                    }
657                    break;
658    
659                default:
660                    // Do nothing
661                    break;
662            }
663    
664            return result;
665        }

667        public int getIndex() {
668            int result;
669    
670            switch (getOpcode()) {
671                case 0x13: // ldc_w
672                case 0x14: // ldc2_w
673                case 0xb2: // getstatic
674                case 0xb3: // putstatic
675                case 0xb4: // getfield
676                case 0xb5: // putfield
677                case 0xb6: // invokevirtual
678                case 0xb7: // invokespecial
679                case 0xb8: // invokestatic
680                case 0xb9: // invokeinterface
681                case 0xba: // invokedynamic
682                case 0xbb: // new
683                case 0xbd: // anewarray
684                case 0xc0: // checkcast
685                case 0xc1: // instanceof
686                case 0xc5: // multianewarray
687                    result = ((getBytecode()[getStart()+1] & 0xff) << 8) | (getBytecode()[getStart()+2] & 0xff);
688                    break;
689                case 0x12: // ldc
690                case 0x15: // iload
691                case 0x16: // llload
692                case 0x17: // fload
693                case 0x18: // dload
694                case 0x19: // aload
695                case 0x36: // istore
696                case 0x37: // lstore
697                case 0x38: // fstore
698                case 0x39: // dstore
699                case 0x3a: // astore
700                case 0x84: // iinc
701                case 0xa9: // ret
702                    result = getBytecode()[getStart()+1] & 0xff;
703                    break;
704                case 0xc4: // wide
705                    result = ((getBytecode()[getStart()+2] & 0xff) << 8) | (getBytecode()[getStart()+3] & 0xff);
706                    break;
707                default:
708                    result = -1;
709                    break;
710            }
711    
712            return result;
713        }

716            ConstantPoolEntry result;
717    
718            switch (getOpcode()) {
719                case 0x12: // ldc
720                case 0x13: // ldc_w
721                case 0x14: // ldc2_w
722                case 0xb2: // getstatic
723                case 0xb3: // putstatic
724                case 0xb4: // getfield
725                case 0xb5: // putfield
726                case 0xb6: // invokevirtual
727                case 0xb7: // invokespecial
728                case 0xb8: // invokestatic
729                case 0xb9: // invokeinterface
730                case 0xba: // invokedynamic
731                case 0xbb: // new
732                case 0xbd: // anewarray
733                case 0xc0: // checkcast
734                case 0xc1: // instanceof
735                case 0xc5: // multianewarray

738                default:

740                    break;

746        public int hashCode() {
747            int result = getOpcode();
748    
749            if (getIndexedConstantPoolEntry() != null) {
750                result ^= getIndexedConstantPoolEntry().hashCode();
751            } else {
752                for (int i=1; i<getLength(); i++) {
753                    result ^= bytecode[start+i];
754                }
755            }
756    
757            return result;
758        }

760        public boolean equals(Object object) {
761            boolean result;
762    
763            if (this == object) {
764                result = true;
765            } else if (object == null || getClass() != object.getClass()) {
766                result = false;
767            } else {
768                Instruction other = (Instruction) object;
769                result = getOpcode() == other.getOpcode();
770    
771                ConstantPoolEntry thisEntry = (getCode() != null) ? getIndexedConstantPoolEntry() : null;
772                ConstantPoolEntry otherEntry = (other.getCode() != null) ? other.getIndexedConstantPoolEntry() : null;
773    
774                if (result && thisEntry != null && otherEntry != null) {
775                    result = thisEntry.equals(otherEntry);
776                } else {
777                    for (int i=1; result && i<getLength(); i++) {
778                        result = bytecode[start+i] == other.bytecode[other.start+i];
779                    }
780                }
781            }
782    
783            return result;
784        }
