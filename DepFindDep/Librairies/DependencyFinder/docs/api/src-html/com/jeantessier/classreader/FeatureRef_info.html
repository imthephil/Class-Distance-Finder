033    package com.jeantessier.classreader;
034    
035    import java.io.*;
036    
037    public abstract class FeatureRef_info extends ConstantPoolEntry {
038        private int classIndex;
039        private int nameAndTypeIndex;
040    
041        public FeatureRef_info(ConstantPool constantPool, DataInputStream in) throws IOException {
042            super(constantPool);
043    
044            classIndex = in.readUnsignedShort();
045            nameAndTypeIndex = in.readUnsignedShort();
046        }
047    
048        public int getClassIndex() {
049            return classIndex;
050        }
051    
052        public Class_info getRawClass() {
053            return (Class_info) getConstantPool().get(getClassIndex());
054        }
055    
056        public String getClassName() {
057            return getRawClass().toString();
058        }
059    
060        public String getClassSimpleName() {
061            return getRawClass().getSimpleName();
062        }
063    
064        public int getNameAndTypeIndex() {
065            return nameAndTypeIndex;
066        }
067    
068        public NameAndType_info getRawNameAndType() {
069            return (NameAndType_info) getConstantPool().get(getNameAndTypeIndex());
070        }
071    
072        public String getNameAndType() {
073            StringBuffer result = new StringBuffer();
074    
075            NameAndType_info nat = getRawNameAndType();
076    
077            result.append(nat.getName()).append(nat.getType());
078    
079            return result.toString();
080        }
081    
082        public abstract String getName();
083    
084        public String getFullName() {
085            return getClassName() + "." + getName();
086        }
087    
088        public abstract String getSignature();
089    
090        public String getFullSignature() {
091            return getClassName() + "." + getSignature();
092        }
093    
094        public String toString() {
095            StringBuffer result = new StringBuffer();
096    
097            Class_info       c   = getRawClass();
098            NameAndType_info nat = getRawNameAndType();
099    
100            result.append(c).append(".").append(nat.getName()).append(nat.getType());
101    
102            return result.toString();
103        }
104    
105        public int hashCode() {
106            return getRawClass().hashCode() ^ getRawNameAndType().hashCode();
107        }
108    
109        public boolean equals(Object object) {
110            boolean result = false;
111    
112            if (this == object) {
113                result = true;
114            } else if (object != null && this.getClass().equals(object.getClass())) {
115                FeatureRef_info other = (FeatureRef_info) object;
116                result = this.getRawClass().equals(other.getRawClass()) && this.getRawNameAndType().equals(other.getRawNameAndType());
117            }
118    
119            return result;
120        }
121    }
