033    package com.jeantessier.dependency;
034    
035    import java.util.*;
036    
037    public class SelectiveTraversalStrategy implements TraversalStrategy {
038        private SelectionCriteria scopeCriteria;
039        private SelectionCriteria filterCriteria;
040    
041        private boolean preOutboundTraversal  = true;
042        private boolean preInboundTraversal   = true;
043        private boolean postOutboundTraversal = false;
044        private boolean postInboundTraversal  = false;
045    
046        public SelectiveTraversalStrategy() {
047            this(new ComprehensiveSelectionCriteria(), new ComprehensiveSelectionCriteria());
048        }
049        
050        public SelectiveTraversalStrategy(SelectionCriteria scopeCriteria, SelectionCriteria filterCriteria) {
051            this.scopeCriteria  = scopeCriteria;
052            this.filterCriteria = filterCriteria;
053        }
054        
055        public boolean doPreOutboundTraversal() {
056            return preOutboundTraversal;
057        }
058    
059        public void setPreOutboundTraversal(boolean preOutboundTraversal) {
060            this.preOutboundTraversal = preOutboundTraversal;
061        }
062    
063        public boolean doPreInboundTraversal() {
064            return preInboundTraversal;
065        }
066    
067        public void setPreInboundTraversal(boolean preInboundTraversal) {
068            this.preInboundTraversal = preInboundTraversal;
069        }
070    
071        public boolean doPostOutboundTraversal() {
072            return postOutboundTraversal;
073        }
074    
075        public void setPostOutboundTraversal(boolean postOutboundTraversal) {
076            this.postOutboundTraversal = postOutboundTraversal;
077        }
078    
079        public boolean doPostInboundTraversal() {
080            return postInboundTraversal;
081        }
082    
083        public void setPostInboundTraversal(boolean postInboundTraversal) {
084            this.postInboundTraversal = postInboundTraversal;
085        }
086    
087        public boolean isInScope(PackageNode node) {
088            return scopeCriteria.matches(node);
089        }
090        
091        public boolean isInScope(ClassNode node) {
092            return scopeCriteria.matches(node);
093        }
094        
095        public boolean isInScope(FeatureNode node) {
096            return scopeCriteria.matches(node);
097        }
098    
099        public boolean isInFilter(PackageNode node) {
100            return filterCriteria.matches(node);
101        }
102        
103        public boolean isInFilter(ClassNode node) {
104            return filterCriteria.matches(node);
105        }
106        
107        public boolean isInFilter(FeatureNode node) {
108            return filterCriteria.matches(node);
109        }
110    
111        public Collection<? extends Node> order(Collection<? extends Node> collection) {
112            return new ArrayList<Node>(collection);
113        }
114    }
