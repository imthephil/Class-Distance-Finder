001    package com.jeantessier.dependencyfinder.cli;
002    
003    import java.io.*;
004    import java.util.*;
005    
006    import com.jeantessier.commandline.*;
007    import com.jeantessier.commandline.Printer;
008    import com.jeantessier.commandline.TextPrinter;
009    import com.jeantessier.dependencyfinder.*;
010    import com.jeantessier.dependency.*;
011    import org.apache.log4j.*;
012    
013    public abstract class Command {
014        public static final String DEFAULT_LOGFILE = "System.out";
015        public static final String DEFAULT_INCLUDES = "//";
016    
017        private String name;
018        private CommandLine commandLine;
019        private CommandLineUsage commandLineUsage;
020    
021        private Date startTime;
022        private VerboseListener verboseListener;
023        protected PrintWriter out;
024    
025        public Command(String name) throws CommandLineException {
026            this.name = name;
027        }
028    
029        public String getName() {
030            return name;
031        }
032    
033        private void resetCommandLine() {
034            commandLine = new CommandLine(getParameterStrategy());
035            populateCommandLineSwitches();
036        }
037    
038        protected ParameterStrategy getParameterStrategy() {
039            return new CollectingParameterStrategy();
040        }
041    
042        protected CommandLine getCommandLine() {
043            if (commandLine == null) {
044                resetCommandLine();
045            }
046    
047            return commandLine;
048        }
049    
050        public CommandLineUsage getCommandLineUsage() {
051            if (commandLineUsage == null) {
052                commandLineUsage = new CommandLineUsage(getName());
053                getCommandLine().accept(commandLineUsage);
054            }
055    
056            return commandLineUsage;
057        }
058    
059        protected VerboseListener getVerboseListener() {
060            return verboseListener;
061        }
062    
063        public void run(String[] args) throws Exception {
064            if (validateCommandLine(args, System.err)) {
065                process();
066            } else {
067                System.exit(1);
068            }
069        }
070    
071        protected void populateCommandLineSwitches() {
072            getCommandLine().addToggleSwitch("echo");
073            getCommandLine().addToggleSwitch("help");
074            getCommandLine().addSingleValueSwitch("out");
075            getCommandLine().addToggleSwitch("time");
076            getCommandLine().addOptionalValueSwitch("verbose", DEFAULT_LOGFILE);
077            getCommandLine().addToggleSwitch("version");
078        }
079    
080        protected void populateCommandLineSwitchesForXMLOutput(String defaultEncoding, String defaultDTDPrefix) {
081            getCommandLine().addSingleValueSwitch("encoding", defaultEncoding);
082            getCommandLine().addSingleValueSwitch("dtd-prefix", defaultDTDPrefix);
083            getCommandLine().addSingleValueSwitch("indent-text");
084        }
085    
086        protected Collection<CommandLineException> parseCommandLine(String[] args) {
087            resetCommandLine();
088            return getCommandLine().parse(args);
089        }
090    
091        protected boolean validateCommandLine(String[] args, PrintStream out) {
092            boolean result = true;
093    
094            Collection<CommandLineException> exceptions = parseCommandLine(args);
095    
096            if (getCommandLine().getToggleSwitch("version")) {
097                showVersion(out);
098                result = false;
099            }
100    
101            if (getCommandLine().getToggleSwitch("help")) {
102                showError(out);
103                result = false;
104            }
105    
106            if (getCommandLine().getToggleSwitch("echo")) {
107                echo(out);
108                result = false;
109            }
110    
111            if (result) {
112                for (CommandLineException exception : exceptions) {
113                    result = false;
114                    Logger.getLogger(getClass()).error(exception);
115                }
116            }
117    
118            return result;
119        }
120    
121        protected Collection<CommandLineException> validateCommandLineForScoping() {
122            Collection<CommandLineException> exceptions = new ArrayList<CommandLineException>();
123    
124            if (hasScopeRegularExpressionSwitches() && hasScopeListSwitches()) {
125                exceptions.add(new CommandLineException("You can use switches for regular expressions or lists for scope, but not at the same time"));
126            }
127    
128            return exceptions;
129        }
130    
131        protected Collection<CommandLineException> validateCommandLineForFiltering() {
132            Collection<CommandLineException> exceptions = new ArrayList<CommandLineException>();
133    
134            if (hasFilterRegularExpressionSwitches() && hasFilterListSwitches()) {
135                exceptions.add(new CommandLineException("You can use switches for regular expressions or lists for filter, but not at the same time"));
136            }
137    
138            return exceptions;
139        }
140    
141        private void process() throws Exception {
142            startProcessing();
143            doProcessing();
144            stopProcessing();
145        }
146    
147        private void startProcessing() throws IOException {
148            startVerboseListener();
149            startTimer();
150            startOutput();
151        }
152    
153        protected abstract void doProcessing() throws Exception;
154    
155        private void stopProcessing() {
156            stopTimer();
157            stopOutput();
158            stopVerboseListener();
159        }
160    
161        private void startVerboseListener() throws IOException {
162            verboseListener = new VerboseListener();
163            if (commandLine.isPresent("verbose")) {
164                if (DEFAULT_LOGFILE.equals(commandLine.getOptionalSwitch("verbose"))) {
165                    verboseListener.setWriter(new OutputStreamWriter(System.out));
166                } else {
167                    verboseListener.setWriter(new FileWriter(commandLine.getOptionalSwitch("verbose")));
168                }
169            }
170        }
171    
172        private void stopVerboseListener() {
173            verboseListener.close();
174        }
175    
176        private void startTimer() {
177            startTime = new Date();
178        }
179    
180        private void stopTimer() {
181            if (commandLine.getToggleSwitch("time")) {
182                Date end = new Date();
183                System.err.println(getClass().getName() + ": " + ((end.getTime() - (double) startTime.getTime()) / 1000) + " secs.");
184            }
185        }
186    
187        private void startOutput() throws IOException {
188            if (getCommandLine().isPresent("out")) {
189                out = new PrintWriter(new FileWriter(getCommandLine().getSingleSwitch("out")));
190            } else {
191                out = new PrintWriter(new OutputStreamWriter(System.out));
192            }
193        }
194    
195        private void stopOutput() {
196            out.close();
197        }
198    
199        protected void echo() {
200            echo(System.err);
201        }
202    
203        protected void echo(PrintStream out) {
204            Printer printer = new TextPrinter(getClass().getSimpleName());
205            getCommandLine().accept(printer);
206            out.println(printer);
207        }
208    
209        protected void showError() {
210            showError(System.err);
211        }
212    
213        protected void showError(PrintStream out) {
214            out.println(getCommandLineUsage());
215            showSpecificUsage(out);
216        }
217    
218        protected void showError(String msg) {
219            showError(System.err, msg);
220        }
221    
222        protected void showError(PrintStream out, String msg) {
223            out.println(msg);
224            showError(out);
225        }
226    
227        protected abstract void showSpecificUsage(PrintStream out);
228    
229        protected void showVersion() {
230            showVersion(System.err);
231        }
232    
233        protected void showVersion(PrintStream out) {
234            Version version = new Version();
235    
236            out.print(version.getImplementationTitle());
237            out.print(" ");
238            out.print(version.getImplementationVersion());
239            out.print(" (c) ");
240            out.print(version.getCopyrightDate());
241            out.print(" ");
242            out.print(version.getCopyrightHolder());
243            out.println();
244    
245            out.print(version.getImplementationURL());
246            out.println();
247    
248            out.print("Compiled on ");
249            out.print(version.getImplementationDate());
250            out.println();
251        }
252    
253        protected void populateCommandLineSwitchesForScoping() {
254            populateRegularExpressionCommandLineSwitches("scope", true, DEFAULT_INCLUDES);
255            populateListCommandLineSwitches("scope");
256        }
257    
258        protected void populateCommandLineSwitchesForFiltering() {
259            populateRegularExpressionCommandLineSwitches("filter", true, DEFAULT_INCLUDES);
260            populateListCommandLineSwitches("filter");
261        }
262    
263        protected void populateCommandLineSwitchesForStartCondition() {
264            populateRegularExpressionCommandLineSwitches("start", false, DEFAULT_INCLUDES);
265            populateListCommandLineSwitches("start");
266        }
267    
268        protected void populateCommandLineSwitchesForStopCondition() {
269            populateRegularExpressionCommandLineSwitches("stop", false, null);
270            populateListCommandLineSwitches("stop");
271        }
272    
273        protected void populateRegularExpressionCommandLineSwitches(String name, boolean addToggles, String defaultIncludes) {
274            if (defaultIncludes != null) {
275                getCommandLine().addMultipleValuesSwitch(name + "-includes", defaultIncludes);
276            } else {
277                getCommandLine().addMultipleValuesSwitch(name + "-includes");
278            }
279            getCommandLine().addMultipleValuesSwitch(name + "-excludes");
280            getCommandLine().addMultipleValuesSwitch("package-" + name + "-includes");
281            getCommandLine().addMultipleValuesSwitch("package-" + name + "-excludes");
282            getCommandLine().addMultipleValuesSwitch("class-" + name + "-includes");
283            getCommandLine().addMultipleValuesSwitch("class-" + name + "-excludes");
284            getCommandLine().addMultipleValuesSwitch("feature-" + name + "-includes");
285            getCommandLine().addMultipleValuesSwitch("feature-" + name + "-excludes");
286    
287            if (addToggles) {
288                getCommandLine().addToggleSwitch("package-" + name);
289                getCommandLine().addToggleSwitch("class-" + name);
290                getCommandLine().addToggleSwitch("feature-" + name);
291            }
292        }
293    
294        protected void populateListCommandLineSwitches(String name) {
295            getCommandLine().addMultipleValuesSwitch(name + "-includes-list");
296            getCommandLine().addMultipleValuesSwitch(name + "-excludes-list");
297        }
298    
299        protected SelectionCriteria getScopeCriteria() {
300            return getSelectionCriteria("scope", new ComprehensiveSelectionCriteria());
301        }
302    
303        protected SelectionCriteria getFilterCriteria() {
304            return getSelectionCriteria("filter", new ComprehensiveSelectionCriteria());
305        }
306    
307        protected SelectionCriteria getStartCriteria() {
308            return getSelectionCriteria("start", new ComprehensiveSelectionCriteria());
309        }
310    
311        protected SelectionCriteria getStopCriteria() {
312            return getSelectionCriteria("stop", new NullSelectionCriteria());
313        }
314    
315        protected SelectionCriteria getSelectionCriteria(String name, SelectionCriteria defaultSelectionCriteria) {
316            SelectionCriteria result = defaultSelectionCriteria;
317    
318            if (hasRegularExpressionSwitches(name)) {
319                RegularExpressionSelectionCriteria regularExpressionFilterCriteria = new RegularExpressionSelectionCriteria();
320    
321                if (getCommandLine().isPresent("package-" + name) || getCommandLine().isPresent("class-" + name) || getCommandLine().isPresent("feature-" + name)) {
322                    regularExpressionFilterCriteria.setMatchingPackages(getCommandLine().getToggleSwitch("package-" + name));
323                    regularExpressionFilterCriteria.setMatchingClasses(getCommandLine().getToggleSwitch("class-" + name));
324                    regularExpressionFilterCriteria.setMatchingFeatures(getCommandLine().getToggleSwitch("feature-" + name));
325                }
326    
327                if (getCommandLine().isPresent(name + "-includes") || (!getCommandLine().isPresent("package-" + name + "-includes") && !getCommandLine().isPresent("class-" + name + "-includes") && !getCommandLine().isPresent("feature-" + name + "-includes"))) {
328                    // Only use the default if nothing else has been specified.
329                    regularExpressionFilterCriteria.setGlobalIncludes(getCommandLine().getMultipleSwitch(name + "-includes"));
330                }
331                regularExpressionFilterCriteria.setGlobalExcludes(getCommandLine().getMultipleSwitch(name + "-excludes"));
332                regularExpressionFilterCriteria.setPackageIncludes(getCommandLine().getMultipleSwitch("package-" + name + "-includes"));
333                regularExpressionFilterCriteria.setPackageExcludes(getCommandLine().getMultipleSwitch("package-" + name + "-excludes"));
334                regularExpressionFilterCriteria.setClassIncludes(getCommandLine().getMultipleSwitch("class-" + name + "-includes"));
335                regularExpressionFilterCriteria.setClassExcludes(getCommandLine().getMultipleSwitch("class-" + name + "-excludes"));
336                regularExpressionFilterCriteria.setFeatureIncludes(getCommandLine().getMultipleSwitch("feature-" + name + "-includes"));
337                regularExpressionFilterCriteria.setFeatureExcludes(getCommandLine().getMultipleSwitch("feature-" + name + "-excludes"));
338    
339                result = regularExpressionFilterCriteria;
340            } else if (hasListSwitches(name)) {
341                result = createCollectionSelectionCriteria(getCommandLine().getMultipleSwitch(name + "-includes-list"), getCommandLine().getMultipleSwitch(name + "-excludes-list"));
342            }
343            
344            return result;
345        }
346    
347        protected boolean hasScopeRegularExpressionSwitches() {
348            return hasRegularExpressionSwitches("scope");
349        }
350    
351        protected boolean hasFilterRegularExpressionSwitches() {
352            return hasRegularExpressionSwitches("filter");
353        }
354    
355        protected boolean hasRegularExpressionSwitches(String name) {
356            Collection<String> switches = getCommandLine().getPresentSwitches();
357    
358            return
359                switches.contains(name + "-includes") ||
360                switches.contains(name + "-excludes") ||
361                switches.contains("package-" + name) ||
362                switches.contains("package-" + name + "-includes") ||
363                switches.contains("package-" + name + "-excludes") ||
364                switches.contains("class-" + name) ||
365                switches.contains("class-" + name + "-includes") ||
366                switches.contains("class-" + name + "-excludes") ||
367                switches.contains("feature-" + name) ||
368                switches.contains("feature-" + name + "-includes") ||
369                switches.contains("feature-" + name + "-excludes");
370        }
371    
372        protected boolean hasScopeListSwitches() {
373            return hasListSwitches("scope");
374        }
375    
376        protected boolean hasFilterListSwitches() {
377            return hasListSwitches("filter");
378        }
379    
380        protected boolean hasListSwitches(String name) {
381            Collection<String> switches = getCommandLine().getPresentSwitches();
382    
383            return
384                switches.contains(name + "-includes-list") ||
385                switches.contains(name + "-excludes-list");
386        }
387    
388        protected CollectionSelectionCriteria createCollectionSelectionCriteria(Collection<String> includes, Collection<String> excludes) {
389            return new CollectionSelectionCriteria(loadCollection(includes), loadCollection(excludes));
390        }
391    
392        private Collection<String> loadCollection(Collection<String> filenames) {
393            Collection<String> result = null;
394    
395            if (!filenames.isEmpty()) {
396                result = new HashSet<String>();
397    
398                for (String filename : filenames) {
399                    BufferedReader reader = null;
400                    try {
401                        reader = new BufferedReader(new FileReader(filename));
402    
403                        String line;
404                        while ((line = reader.readLine()) != null) {
405                            result.add(line);
406                        }
407                    } catch (IOException ex) {
408                        Logger.getLogger(getClass()).error("Couldn't read file " + filename, ex);
409                    } finally {
410                        try {
411                            if (reader != null) {
412                                reader.close();
413                            }
414                        } catch (IOException ex) {
415                            Logger.getLogger(getClass()).error("Couldn't close file " + filename, ex);
416                        }
417                    }
418                }
419            }
420    
421            return result;
422        }
