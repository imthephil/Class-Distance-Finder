033    package com.jeantessier.dependencyfinder.cli;
034    
035    import com.jeantessier.classreader.*;
036    import com.jeantessier.commandline.*;
037    import com.jeantessier.diff.*;
038    import org.apache.log4j.*;
039    
040    public class ClassClassDiff extends DiffCommand {
041        public ClassClassDiff() throws CommandLineException {
042            super("ClassClassDiff");
043        }
044    
045        public void doProcessing() throws Exception {
046            // Collecting data, first classfiles from JARs,
047            // then package/class trees using NodeFactory.
048    
049            ClassfileLoader oldJar = new AggregatingClassfileLoader();
050            oldJar.addLoadListener(getVerboseListener());
051            oldJar.load(getCommandLine().getMultipleSwitch("old"));
052    
053            ClassfileLoader newJar = new AggregatingClassfileLoader();
054            newJar.addLoadListener(getVerboseListener());
055            newJar.load(getCommandLine().getMultipleSwitch("new"));
056    
057            DifferenceStrategy baseStrategy = getBaseStrategy(getCommandLine().getToggleSwitch("code"));
058            DifferenceStrategy strategy = getStrategy(getCommandLine().getSingleSwitch("level"), baseStrategy);
059    
060            if (getCommandLine().isPresent("filter")) {
061                strategy = new ListBasedDifferenceStrategy(strategy, getCommandLine().getSingleSwitch("filter"));
062            }
063    
064            // Starting to compare, first at class level,
065            // then descending to feature level.
066    
067            Logger.getLogger(getClass()).info("Comparing ...");
068            getVerboseListener().print("Comparing ...");
069    
070            String name = getCommandLine().getSingleSwitch("name");
071            String oldLabel = getCommandLine().isPresent("old-label") ? getCommandLine().getSingleSwitch("old-label") : getCommandLine().getMultipleSwitch("old").toString();
072            String newLabel = getCommandLine().isPresent("new-label") ? getCommandLine().getSingleSwitch("new-label") : getCommandLine().getMultipleSwitch("new").toString();
073    
074            Classfile oldClass = oldJar.getAllClassfiles().iterator().next();
075            Classfile newClass = newJar.getAllClassfiles().iterator().next();
076    
077            DifferencesFactory factory = new DifferencesFactory(strategy);
078            Differences differences = factory.createClassDifferences(name, oldClass, newClass);
079    
080            Logger.getLogger(getClass()).info("Printing results ...");
081            getVerboseListener().print("Printing results ...");
082    
083            Report report = new Report(getCommandLine().getSingleSwitch("encoding"), getCommandLine().getSingleSwitch("dtd-prefix"));
084            if (getCommandLine().isPresent("indent-text")) {
085                report.setIndentText(getCommandLine().getSingleSwitch("indent-text"));
086            }
087            report.setName(name);
088            report.setOldVersion(oldLabel);
089            report.setNewVersion(newLabel);
090    
091            differences.accept(report);
092            out.print(report);
093        }
094    
095        public static void main(String[] args) throws Exception {
096            new ClassClassDiff().run(args);
097        }
098    }
