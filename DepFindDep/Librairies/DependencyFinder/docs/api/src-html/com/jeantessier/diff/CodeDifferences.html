033    package com.jeantessier.diff;
034    
035    import com.jeantessier.classreader.*;
036    
037    /**
038     * Documents the difference, if any, for a given method or superclass.
039     * This class checks the Code attribute (if any) in addition to anything
040     * checked by superclasses.
041     */
042    public abstract class CodeDifferences extends FeatureDifferences {
043        private boolean codeDifference = false;
044    
045        protected CodeDifferences(String name, Method_info oldFeature, Method_info newFeature) {
046            super(name, oldFeature, newFeature);
047        }
048    
049        public boolean isCodeDifference() {
050            return codeDifference;
051        }
052    
053        /**
054         * Only the DifferencesFactory can set this flag
055         */
056        void setCodeDifference(boolean codeDifference) {
057            this.codeDifference = codeDifference;
058        }
059    
060        public boolean isModified() {
061            return isCodeDifference() || super.isModified();
062        }
063    }
