241        // Classfile
242        public void visitClassfile(Classfile classfile) {
243            if ((classfile.getAccessFlag() & Classfile.ACC_PUBLIC) != 0) {
244                publicClasses.add(classfile);
245            } else {
246                packageClasses.add(classfile);
247            }
248    
249            if ((classfile.getAccessFlag() & Classfile.ACC_FINAL) != 0) {
250                finalClasses.add(classfile);
251            }
252    
253            if ((classfile.getAccessFlag() & Classfile.ACC_INTERFACE) != 0) {
254                interfaces.add(classfile);
255            } else {
256                classes.add(classfile);
257            }
258    
259            if ((classfile.getAccessFlag() & Classfile.ACC_ABSTRACT) != 0) {
260                abstractClasses.add(classfile);
261            }
262    
263            super.visitClassfile(classfile);
264        }
265    
266        // Features
267        public void visitField_info(Field_info entry) {
268            fields.add(entry);
269    
270            if ((entry.getAccessFlag() & Field_info.ACC_PUBLIC) != 0) {
271                publicFields.add(entry);
272            } else if ((entry.getAccessFlag() & Field_info.ACC_PRIVATE) != 0) {
273                privateFields.add(entry);
274            } else if ((entry.getAccessFlag() & Field_info.ACC_PROTECTED) != 0) {
275                protectedFields.add(entry);
276            } else {
277                packageFields.add(entry);
278            }
279    
280            if ((entry.getAccessFlag() & Field_info.ACC_STATIC) != 0) {
281                staticFields.add(entry);
282            }
283    
284            if ((entry.getAccessFlag() & Field_info.ACC_FINAL) != 0) {
285                finalFields.add(entry);
286            }
287    
288            if ((entry.getAccessFlag() & Field_info.ACC_VOLATILE) != 0) {
289                volatileFields.add(entry);
290            }
291    
292            if ((entry.getAccessFlag() & Field_info.ACC_TRANSIENT) != 0) {
293                transientFields.add(entry);
294            }
295    
296            super.visitField_info(entry);
297        }
298

335        // Attributes
336        public void visitSynthetic_attribute(Synthetic_attribute attribute) {
337            Object owner = attribute.getOwner();
338        
339            if (owner instanceof Classfile) {
340                syntheticClasses.add(owner);
341            } else if (owner instanceof Field_info) {
342                syntheticFields.add(owner);
343            } else if (owner instanceof Method_info) {
344                syntheticMethods.add(owner);
345            } else {
346                Logger.getLogger(getClass()).warn("Synthetic attribute on unknown Visitable: " + owner.getClass().getName());
347            }
348        }
349    
350        public void visitDeprecated_attribute(Deprecated_attribute attribute) {
351            Object owner = attribute.getOwner();
352        
353            if (owner instanceof Classfile) {
354                deprecatedClasses.add(owner);
355            } else if (owner instanceof Field_info) {
356                deprecatedFields.add(owner);
357            } else if (owner instanceof Method_info) {
358                deprecatedMethods.add(owner);
359            } else {
360                Logger.getLogger(getClass()).warn("Deprecated attribute on unknown Visitable: " + owner.getClass().getName());
361            }
362        }
363    
364        public void visitCustom_attribute(Custom_attribute attribute) {
365            customAttributes.add(attribute);
366        }
367    
368        // Attribute helpers
369        public void visitInstruction(Instruction helper) {
370            getInstructionCounts()[helper.getOpcode()]++;
371    
372            super.visitInstruction(helper);
373        }
374    
375        public void visitInnerClass(InnerClass helper) {
376            if ((helper.getAccessFlag() & InnerClass.ACC_PUBLIC) != 0) {
377                publicInnerClasses.add(helper);
378            } else if ((helper.getAccessFlag() & InnerClass.ACC_PRIVATE) != 0) {
379                privateInnerClasses.add(helper);
380            } else if ((helper.getAccessFlag() & InnerClass.ACC_PROTECTED) != 0) {
381                protectedInnerClasses.add(helper);
382            } else {
383                packageInnerClasses.add(helper);
384            }
385    
386            if ((helper.getAccessFlag() & InnerClass.ACC_STATIC) != 0) {
387                staticInnerClasses.add(helper);
388            }
389    
390            if ((helper.getAccessFlag() & InnerClass.ACC_FINAL) != 0) {
391                finalInnerClasses.add(helper);
392            }
393    
394            if ((helper.getAccessFlag() & InnerClass.ACC_INTERFACE) != 0) {
395                interfaces.add(helper);
396            } else {
397                classes.add(helper);
398            }
399    
400            if ((helper.getAccessFlag() & InnerClass.ACC_ABSTRACT) != 0) {
401                abstractInnerClasses.add(helper);
402            }
403        }
