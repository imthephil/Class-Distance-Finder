033    package com.jeantessier.classreader;
034    
035    import java.io.*;
036    
037    public class Class_info extends ConstantPoolEntry {
038        private int nameIndex;
039    
040        public Class_info(ConstantPool constantPool, DataInputStream in) throws IOException {
041            super(constantPool);
042    
043            nameIndex = in.readUnsignedShort();
044        }
045    
046        public int getNameIndex() {
047            return nameIndex;
048        }
049    
050        public UTF8_info getRawName() {
051            return (UTF8_info) getConstantPool().get(getNameIndex());
052        }
053    
054        public String getName() {
055            return SignatureHelper.convertClassName(getRawName().toString());
056        }
057    
058        public String getSimpleName() {
059            return getName().substring(getName().lastIndexOf(".") + 1);
060        }
061    
062        public String toString() {
063            return getName();
064        }
065    
066        public int hashCode() {
067            return getRawName().hashCode();
068        }
069    
070        public boolean equals(Object object) {
071            boolean result = false;
072    
073            if (this == object) {
074                result = true;
075            } else if (object != null && this.getClass().equals(object.getClass())) {
076                Class_info other = (Class_info) object;
077                result = this.getRawName().equals(other.getRawName());
078            }
079    
080            return result;
081        }
082    
083        public void accept(Visitor visitor) {
084            visitor.visitClass_info(this);
085        }
086    }
