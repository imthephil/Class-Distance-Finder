033    package com.jeantessier.dependencyfinder.cli;
034    
035    import java.io.*;
036    import java.util.*;
037    
038    import com.jeantessier.commandline.*;
039    import com.jeantessier.diff.*;
040    
041    public class ListDiff extends Command {
042        public ListDiff() throws CommandLineException {
043            super("ListDiff");
044        }
045    
046        protected void showSpecificUsage(PrintStream out) {
047            out.println();
048            out.println("Defaults is text output to the console.");
049            out.println();
050        }
051    
052        public static void main(String[] args) throws Exception {
053            new ListDiff().run(args);
054        }
055    
056        protected void populateCommandLineSwitches() {
057            super.populateCommandLineSwitches();
058            populateCommandLineSwitchesForXMLOutput(ListDiffPrinter.DEFAULT_ENCODING,  ListDiffPrinter.DEFAULT_DTD_PREFIX);
059    
060            getCommandLine().addSingleValueSwitch("name");
061            getCommandLine().addSingleValueSwitch("old-label");
062            getCommandLine().addSingleValueSwitch("old", true);
063            getCommandLine().addSingleValueSwitch("new-label");
064            getCommandLine().addSingleValueSwitch("new", true);
065            getCommandLine().addToggleSwitch("compress");
066        }
067    
068        protected void doProcessing() throws Exception {
069            String line;
070            
071            getVerboseListener().print("Loading old list ...");
072            Collection<String> oldAPI = new TreeSet<String>();
073            BufferedReader oldIn = new BufferedReader(new FileReader(getCommandLine().getSingleSwitch("old")));
074            while((line = oldIn.readLine()) != null) {
075                oldAPI.add(line);
076            }
077            
078            getVerboseListener().print("Loading new list ...");
079            Collection<String> newAPI = new TreeSet<String>();
080            BufferedReader newIn = new BufferedReader(new FileReader(getCommandLine().getSingleSwitch("new")));
081            while((line = newIn.readLine()) != null) {
082                newAPI.add(line);
083            }
084            
085            ListDiffPrinter printer = new ListDiffPrinter(getCommandLine().getToggleSwitch("compress"), getCommandLine().getSingleSwitch("encoding"), getCommandLine().getSingleSwitch("dtd-prefix"));
086            printer.setName(getCommandLine().getSingleSwitch("name"));
087            printer.setOldVersion(getCommandLine().getSingleSwitch("old-label"));
088            printer.setNewVersion(getCommandLine().getSingleSwitch("new-label"));
089            if (getCommandLine().isPresent("indent-text")) {
090                printer.setIndentText(getCommandLine().getSingleSwitch("indent-text"));
091            }
092    
093            getVerboseListener().print("Computing removed elements ...");
094            for (String name : oldAPI) {
095                if (!newAPI.contains(name)) {
096                    printer.remove(name);
097                }
098            }
099    
100            getVerboseListener().print("Computing added elements ...");
101            for (String name : newAPI) {
102                if (!oldAPI.contains(name)) {
103                    printer.add(name);
104                }
105            }
106    
107            getVerboseListener().print("Printing results ...");
108            out.print(printer);
109        }
110    }
