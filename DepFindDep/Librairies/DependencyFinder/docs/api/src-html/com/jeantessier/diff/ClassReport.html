039    public class ClassReport extends Printer implements Comparable {
040        private ClassDifferences differences;
041    
042        private Collection<FeatureDifferences> removedFields = new TreeSet<FeatureDifferences>();
043        private Collection<FeatureDifferences> removedConstructors = new TreeSet<FeatureDifferences>();
044        private Collection<FeatureDifferences> removedMethods = new TreeSet<FeatureDifferences>();
045    
046        private Collection<FeatureDifferences> deprecatedFields = new TreeSet<FeatureDifferences>();
047        private Collection<FeatureDifferences> deprecatedConstructors = new TreeSet<FeatureDifferences>();
048        private Collection<FeatureDifferences> deprecatedMethods = new TreeSet<FeatureDifferences>();
049    
050        private Collection<FieldDifferences> modifiedFields = new TreeSet<FieldDifferences>();
051        private Collection<CodeDifferences> modifiedConstructors = new TreeSet<CodeDifferences>();
052        private Collection<CodeDifferences> modifiedMethods = new TreeSet<CodeDifferences>();
053    
054        private Collection<FeatureDifferences> undeprecatedFields = new TreeSet<FeatureDifferences>();
055        private Collection<FeatureDifferences> undeprecatedConstructors = new TreeSet<FeatureDifferences>();
056        private Collection<FeatureDifferences> undeprecatedMethods = new TreeSet<FeatureDifferences>();
057    
058        private Collection<FeatureDifferences> newFields = new TreeSet<FeatureDifferences>();
059        private Collection<FeatureDifferences> newConstructors = new TreeSet<FeatureDifferences>();
060        private Collection<FeatureDifferences> newMethods = new TreeSet<FeatureDifferences>();
061    
062        public void visitClassDifferences(ClassDifferences differences) {
063            this.differences = differences;
064    
065            for (Differences featureDifference : differences.getFeatureDifferences()) {
066                featureDifference.accept(this);
067            }
068        }
069    
070        public void visitInterfaceDifferences(InterfaceDifferences differences) {
071            this.differences = differences;
072    
073            for (Differences featureDifference : differences.getFeatureDifferences()) {
074                featureDifference.accept(this);
075            }
076        }
077    
078        public void visitFieldDifferences(FieldDifferences differences) {
079            if (differences.isRemoved()) {
080                removedFields.add(differences);
081            }
082        
083            if (differences.isModified()) {
084                modifiedFields.add(differences);
085            }
086        
087            if (differences.isNew()) {
088                newFields.add(differences);
089            }
090    
091            if (isDeprecated()) {
092                deprecatedFields.add(differences);
093            }
094    
095            if (isUndeprecated()) {
096                undeprecatedFields.add(differences);
097            }
098        }
099    
100        public void visitConstructorDifferences(ConstructorDifferences differences) {
101            if (differences.isRemoved()) {
102                removedConstructors.add(differences);
103            }
104        
105            if (differences.isModified()) {
106                modifiedConstructors.add(differences);
107            }
108        
109            if (differences.isNew()) {
110                newConstructors.add(differences);
111            }
112    
113            if (isDeprecated()) {
114                deprecatedConstructors.add(differences);
115            }
116    
117            if (isUndeprecated()) {
118                undeprecatedConstructors.add(differences);
119            }
120        }
121    
122        public void visitMethodDifferences(MethodDifferences differences) {
123            if (differences.isRemoved()) {
124                removedMethods.add(differences);
125            }
126        
127            if (differences.isModified()) {
128                modifiedMethods.add(differences);
129            }
130        
131            if (differences.isNew()) {
132                newMethods.add(differences);
133            }
134    
135            if (isDeprecated()) {
136                deprecatedMethods.add(differences);
137            }
138    
139            if (isUndeprecated()) {
140                undeprecatedMethods.add(differences);
141            }
142        }

144        public String toString() {
145            raiseIndent();
146            raiseIndent();
147    
148            indent().append("<class>").eol();
149            raiseIndent();
150    
151            indent().append("<name>").append(differences.getName()).append("</name>").eol();
152    
153            if (differences.isDeclarationModified()) {
154                indent().append("<modified-declaration>").eol();
155                raiseIndent();
156    
157                indent().append("<old-declaration").append(breakdownDeclaration(differences.getOldClass())).append(">").append(differences.getOldDeclaration()).append("</old-declaration>").eol();
158                indent().append("<new-declaration").append(breakdownDeclaration(differences.getNewClass())).append(">").append(differences.getNewDeclaration()).append("</new-declaration>").eol();
159    
160                lowerIndent();
161                indent().append("</modified-declaration>").eol();
162            }
163    
164            if (removedFields.size() != 0) {
165                indent().append("<removed-fields>").eol();
166                raiseIndent();
167    
168                for (FeatureDifferences fd : removedFields) {
169                    indent().append("<declaration").append(breakdownDeclaration((Field_info) fd.getOldFeature())).append(fd.isInherited() ? " inherited=\"yes\"" : "").append(">").append(fd.getOldDeclaration()).append("</declaration>").eol();
170                }
171    
172                lowerIndent();
173                indent().append("</removed-fields>").eol();
174            }
175    
176            if (removedConstructors.size() != 0) {
177                indent().append("<removed-constructors>").eol();
178                raiseIndent();
179    
180                for (FeatureDifferences fd : removedConstructors) {
181                    indent().append("<declaration").append(breakdownDeclaration((Method_info) fd.getOldFeature())).append(fd.isInherited() ? " inherited=\"yes\"" : "").append(">").append(fd.getOldDeclaration()).append("</declaration>").eol();
182                }
183    
184                lowerIndent();
185                indent().append("</removed-constructors>").eol();
186            }
187    
188            if (removedMethods.size() != 0) {
189                indent().append("<removed-methods>").eol();
190                raiseIndent();
191    
192                for (FeatureDifferences fd : removedMethods) {
193                    indent().append("<declaration").append(breakdownDeclaration((Method_info) fd.getOldFeature())).append(fd.isInherited() ? " inherited=\"yes\"" : "").append(">").append(fd.getOldDeclaration()).append("</declaration>").eol();
194                }
195    
196                lowerIndent();
197                indent().append("</removed-methods>").eol();
198            }
199    
200            if (deprecatedFields.size() != 0) {
201                indent().append("<deprecated-fields>").eol();
202                raiseIndent();
203    
204                for (FeatureDifferences fd : deprecatedFields) {
205                    indent().append("<declaration").append(breakdownDeclaration((Field_info) fd.getNewFeature())).append(">").append(fd.getOldDeclaration()).append("</declaration>").eol();
206                }
207    
208                lowerIndent();
209                indent().append("</deprecated-fields>").eol();
210            }
211    
212            if (deprecatedConstructors.size() != 0) {
213                indent().append("<deprecated-constructors>").eol();
214                raiseIndent();
215    
216                for (FeatureDifferences fd : deprecatedConstructors) {
217                    indent().append("<declaration").append(breakdownDeclaration((Method_info) fd.getNewFeature())).append(">").append(fd.getOldDeclaration()).append("</declaration>").eol();
218                }
219    
220                lowerIndent();
221                indent().append("</deprecated-constructors>").eol();
222            }
223    
224            if (deprecatedMethods.size() != 0) {
225                indent().append("<deprecated-methods>").eol();
226                raiseIndent();
227    
228                for (FeatureDifferences fd : deprecatedMethods) {
229                    indent().append("<declaration").append(breakdownDeclaration((Method_info) fd.getNewFeature())).append(">").append(fd.getOldDeclaration()).append("</declaration>").eol();
230                }
231    
232                lowerIndent();
233                indent().append("</deprecated-methods>").eol();
234            }
235    
236            if (modifiedFields.size() != 0) {
237                indent().append("<modified-fields>").eol();
238                raiseIndent();
239    
240                for (FieldDifferences fd : modifiedFields) {
241                    indent().append("<feature>").eol();
242                    raiseIndent();
243    
244                    indent().append("<name>").append(fd.getName()).append("</name>").eol();
245    
246                    indent().append("<modified-declaration>").eol();
247                    raiseIndent();
248    
249                    Field_info oldField = (Field_info) fd.getOldFeature();
250                    Field_info newField = (Field_info) fd.getNewFeature();
251                    if (fd.isConstantValueDifference()) {
252                        indent().append("<old-declaration").append(breakdownDeclaration(oldField)).append(">").append(oldField.getFullDeclaration()).append("</old-declaration>").eol();
253                        indent().append("<new-declaration").append(breakdownDeclaration(newField)).append(">").append(newField.getFullDeclaration()).append("</new-declaration>").eol();
254                    } else {
255                        indent().append("<old-declaration").append(breakdownDeclaration(oldField)).append(">").append(oldField.getDeclaration()).append("</old-declaration>").eol();
256                        indent().append("<new-declaration").append(breakdownDeclaration(newField)).append(">").append(newField.getDeclaration()).append("</new-declaration>").eol();
257                    }
258    
259                    lowerIndent();
260                    indent().append("</modified-declaration>").eol();
261    
262                    lowerIndent();
263                    indent().append("</feature>").eol();
264                }
265    
266                lowerIndent();
267                indent().append("</modified-fields>").eol();
268            }
269    
270            if (modifiedConstructors.size() != 0) {
271                indent().append("<modified-constructors>").eol();
272                raiseIndent();
273    
274                for (CodeDifferences cd : modifiedConstructors) {
275                    indent().append("<feature>").eol();
276                    raiseIndent();
277    
278                    indent().append("<name>").append(cd.getName()).append("</name>").eol();
279    
280                    if (!cd.getOldDeclaration().equals(cd.getNewDeclaration())) {
281                        indent().append("<modified-declaration>").eol();
282                        raiseIndent();
283                        indent().append("<old-declaration").append(breakdownDeclaration((Method_info) cd.getOldFeature())).append(">").append(cd.getOldDeclaration()).append("</old-declaration>").eol();
284                        indent().append("<new-declaration").append(breakdownDeclaration((Method_info) cd.getNewFeature())).append(">").append(cd.getNewDeclaration()).append("</new-declaration>").eol();
285                        lowerIndent();
286                        indent().append("</modified-declaration>").eol();
287                    }
288    
289                    if (cd.isCodeDifference()) {
290                        indent().append("<modified-code").append(breakdownDeclaration((Method_info) cd.getNewFeature())).append(">").append(cd.getNewDeclaration()).append("</modified-code>").eol();
291                    }
292    
293                    lowerIndent();
294                    indent().append("</feature>").eol();
295                }
296    
297                lowerIndent();
298                indent().append("</modified-constructors>").eol();
299            }
300    
301            if (modifiedMethods.size() != 0) {
302                indent().append("<modified-methods>").eol();
303                raiseIndent();
304    
305                for (CodeDifferences md : modifiedMethods) {
306                    indent().append("<feature>").eol();
307                    raiseIndent();
308    
309                    indent().append("<name>").append(md.getName()).append("</name>").eol();
310    
311                    if (!md.getOldDeclaration().equals(md.getNewDeclaration())) {
312                        indent().append("<modified-declaration>").eol();
313                        raiseIndent();
314                        indent().append("<old-declaration").append(breakdownDeclaration((Method_info) md.getOldFeature())).append(">").append(md.getOldDeclaration()).append("</old-declaration>").eol();
315                        indent().append("<new-declaration").append(breakdownDeclaration((Method_info) md.getNewFeature())).append(">").append(md.getNewDeclaration()).append("</new-declaration>").eol();
316                        lowerIndent();
317                        indent().append("</modified-declaration>").eol();
318                    }
319    
320                    if (md.isCodeDifference()) {
321                        indent().append("<modified-code").append(breakdownDeclaration((Method_info) md.getNewFeature())).append(">").append(md.getNewDeclaration()).append("</modified-code>").eol();
322                    }
323    
324                    lowerIndent();
325                    indent().append("</feature>").eol();
326                }
327    
328                lowerIndent();
329                indent().append("</modified-methods>").eol();
330            }
331    
332            if (undeprecatedFields.size() != 0) {
333                indent().append("<undeprecated-fields>").eol();
334                raiseIndent();
335    
336                for (FeatureDifferences fd : undeprecatedFields) {
337                    indent().append("<declaration").append(breakdownDeclaration((Field_info) fd.getNewFeature())).append(">").append(fd.getOldDeclaration()).append("</declaration>").eol();
338                }
339    
340                lowerIndent();
341                indent().append("</undeprecated-fields>").eol();
342            }
343    
344            if (undeprecatedConstructors.size() != 0) {
345                indent().append("<undeprecated-constructors>").eol();
346                raiseIndent();
347    
348                for (FeatureDifferences fd : undeprecatedConstructors) {
349                    indent().append("<declaration").append(breakdownDeclaration((Method_info) fd.getNewFeature())).append(">").append(fd.getOldDeclaration()).append("</declaration>").eol();
350                }
351    
352                lowerIndent();
353                indent().append("</undeprecated-constructors>").eol();
354            }
355    
356            if (undeprecatedMethods.size() != 0) {
357                indent().append("<undeprecated-methods>").eol();
358                raiseIndent();
359    
360                for (FeatureDifferences fd : undeprecatedMethods) {
361                    indent().append("<declaration").append(breakdownDeclaration((Method_info) fd.getNewFeature())).append(">").append(fd.getOldDeclaration()).append("</declaration>").eol();
362                }
363    
364                lowerIndent();
365                indent().append("</undeprecated-methods>").eol();
366            }
367    
368            if (newFields.size() != 0) {
369                indent().append("<new-fields>").eol();
370                raiseIndent();
371    
372                for (FeatureDifferences fd : newFields) {
373                    indent().append("<declaration").append(breakdownDeclaration((Field_info) fd.getNewFeature())).append(">").append(fd.getNewDeclaration()).append("</declaration>").eol();
374                }
375    
376                lowerIndent();
377                indent().append("</new-fields>").eol();
378            }
379    
380            if (newConstructors.size() != 0) {
381                indent().append("<new-constructors>").eol();
382                raiseIndent();
383    
384                for (FeatureDifferences fd : newConstructors) {
385                    indent().append("<declaration").append(breakdownDeclaration((Method_info) fd.getNewFeature())).append(">").append(fd.getNewDeclaration()).append("</declaration>").eol();
386                }
387    
388                lowerIndent();
389                indent().append("</new-constructors>").eol();
390            }
391    
392            if (newMethods.size() != 0) {
393                indent().append("<new-methods>").eol();
394                raiseIndent();
395    
396                for (FeatureDifferences fd : newMethods) {
397                    indent().append("<declaration").append(breakdownDeclaration((Method_info) fd.getNewFeature())).append(">").append(fd.getNewDeclaration()).append("</declaration>").eol();
398                }
399    
400                lowerIndent();
401                indent().append("</new-methods>").eol();
402            }
403    
404            lowerIndent();
405            indent().append("</class>").eol();
406    
407            return super.toString();
408        }

410        private String breakdownDeclaration(Classfile element) {
411            StringBuffer result = new StringBuffer();
412    
413            if (element != null) {
414                if (element.isPublic())     result.append(" visibility=\"public\"");
415                if (element.isPackage())    result.append(" visibility=\"package\"");
416                if (element.isFinal())      result.append(" final=\"yes\"");
417                if (element.isSuper())      result.append(" super=\"yes\"");
418                if (element.isSynthetic())  result.append(" synthetic=\"yes\"");
419                if (element.isDeprecated()) result.append(" deprecated=\"yes\"");
420    
421                result.append(" name=\"").append(element.getClassName()).append("\"");
422    
423                if (element.isInterface()) {
424                    result.append(" interface=\"yes\"");
425            
426                    result.append(" extends=\"");
427                    Iterator i = element.getAllInterfaces().iterator();
428                    while (i.hasNext()) {
429                        result.append(i.next());
430                        if (i.hasNext()) {
431                            result.append(", ");
432                        }
433                    }
434                    result.append("\"");
435                } else {
436                    if (element.isAbstract()) result.append(" abstract=\"yes\"");
437            
438                    result.append(" extends=\"").append(element.getSuperclassName()).append("\"");
439            
440                    result.append(" implements=\"");
441                    Iterator i = element.getAllInterfaces().iterator();
442                    while (i.hasNext()) {
443                        result.append(i.next());
444                        if (i.hasNext()) {
445                            result.append(", ");
446                        }
447                    }
448                    result.append("\"");
449                }
450            }
451    
452            return result.toString();
453        }

455        private String breakdownDeclaration(Field_info element) {
456            StringBuffer result = new StringBuffer();
457    
458            if (element != null) {
459                if (element.isPublic())     result.append(" visibility=\"public\"");
460                if (element.isProtected())  result.append(" visibility=\"protected\"");
461                if (element.isPackage())    result.append(" visibility=\"package\"");
462                if (element.isPrivate())    result.append(" visibility=\"private\"");
463                if (element.isStatic())     result.append(" static=\"yes\"");
464                if (element.isFinal())      result.append(" final=\"yes\"");
465                if (element.isVolatile())   result.append(" volatile=\"yes\"");
466                if (element.isTransient())  result.append(" transient=\"yes\"");
467                if (element.isSynthetic())  result.append(" synthetic=\"yes\"");
468                if (element.isDeprecated()) result.append(" deprecated=\"yes\"");
469    
470                result.append(" type=\"").append(element.getType()).append("\"");
471                result.append(" name=\"").append(element.getName()).append("\"");
472                result.append(" signature=\"").append(element.getSignature()).append("\"");
473                result.append(" full-signature=\"").append(element.getFullSignature()).append("\"");
474    
475                if (element.getConstantValue() != null) {
476                    result.append(" value=\"").append(element.getConstantValue().getRawValue()).append("\"");
477                }
478            }
479    
480            return result.toString();
481        }
482    
483        private String breakdownDeclaration(Method_info element) {
484            StringBuffer result = new StringBuffer();
485    
486            if (element != null) {
487                if (element.isPublic())       result.append(" visibility=\"public\"");
488                if (element.isProtected())    result.append(" visibility=\"protected\"");
489                if (element.isPackage())      result.append(" visibility=\"package\"");
490                if (element.isPrivate())      result.append(" visibility=\"private\"");
491                if (element.isStatic())       result.append(" static=\"yes\"");
492                if (element.isFinal())        result.append(" final=\"yes\"");
493                if (element.isSynchronized()) result.append(" synchronized=\"yes\"");
494                if (element.isNative())       result.append(" native=\"yes\"");
495                if (element.isAbstract())     result.append(" abstract=\"yes\"");
496                if (element.isStrict())       result.append(" strict=\"yes\"");
497                if (element.isSynthetic())    result.append(" synthetic=\"yes\"");
498                if (element.isDeprecated())   result.append(" deprecated=\"yes\"");
499    
500                if (!element.getName().equals("<init>") && !element.getName().equals("<clinit>")) {
501                    result.append(" return-type=\"").append(element.getReturnType()).append("\"");
502                }
503    
504                result.append(" signature=\"").append(element.getSignature()).append("\"");
505                result.append(" full-signature=\"").append(element.getFullSignature()).append("\"");
506    
507                result.append(" throws=\"");
508                Iterator i = element.getExceptions().iterator();
509                while (i.hasNext()) {
510                    result.append(i.next());
511                    if (i.hasNext()) {
512                        result.append(", ");
513                    }
514                }
515                result.append("\"");
516            }
517    
518            return result.toString();
519        }
520    
521        public int compareTo(Object other) {
522            int result = 0;
523    
524            if (other instanceof ClassReport) {
525                result = differences.compareTo(((ClassReport) other).differences);
526            } else {
527                throw new ClassCastException("Unable to compare ClassReport to " + other.getClass().getName());
528            }
529    
530            return result;
531        }
