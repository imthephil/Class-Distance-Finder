033    package com.jeantessier.classreader;
034    
035    import java.io.*;
036    
037    import org.apache.log4j.*;
038    
039    public class LocalVariable implements Visitable {
040        private LocalVariableTable_attribute localVariableTable;
041        private int                          startPC;
042        private int                          length;
043        private int                          nameIndex;
044        private int                          descriptorIndex;
045        private int                          index;
046    
047        public LocalVariable(LocalVariableTable_attribute localVariableTable, DataInputStream in) throws IOException {
048            this.localVariableTable = localVariableTable;
049    
050            startPC = in.readUnsignedShort();
051            Logger.getLogger(getClass()).debug("start PC: " + startPC);
052    
053            length = in.readUnsignedShort();
054            Logger.getLogger(getClass()).debug("length: " + length);
055    
056            nameIndex = in.readUnsignedShort();
057            Logger.getLogger(getClass()).debug("name: " + nameIndex + " (" + getName() + ")");
058    
059            descriptorIndex = in.readUnsignedShort();
060            Logger.getLogger(getClass()).debug("descriptor: " + descriptorIndex + " (" + getDescriptor() + ")");
061    
062            index = in.readUnsignedShort();
063            Logger.getLogger(getClass()).debug("index: " + index);
064        }
065    
066        public LocalVariableTable_attribute getLocalVariableTable() {
067            return localVariableTable;
068        }
069    
070        public int getStartPC() {
071            return startPC;
072        }
073    
074        public int getLength() {
075            return length;
076        }
077    
078        public int getNameIndex() {
079            return nameIndex;
080        }
081    
082        public UTF8_info getRawName() {
083            return (UTF8_info) getLocalVariableTable().getClassfile().getConstantPool().get(getNameIndex());
084        }
085    
086        public String getName() {
087            return getRawName().toString();
088        }
089    
090        public int getDescriptorIndex() {
091            return descriptorIndex;
092        }
093    
094        public UTF8_info getRawDescriptor() {
095            return (UTF8_info) getLocalVariableTable().getClassfile().getConstantPool().get(getDescriptorIndex());
096        }
097    
098        public String getDescriptor() {
099            return getRawDescriptor().toString();
100        }
101    
102        public int getIndex() {
103            return index;
104        }
105    
106        public String toString() {
107            return "Local variable " + getDescriptor() + " " + getName();
108        }
109    
110        public void accept(Visitor visitor) {
111            visitor.visitLocalVariable(this);
112        }
113    }
