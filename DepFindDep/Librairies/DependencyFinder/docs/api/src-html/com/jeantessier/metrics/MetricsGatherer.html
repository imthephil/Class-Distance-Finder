119        public void visitClassfiles(Collection<Classfile> classfiles) {
120            fireBeginSession(classfiles.size());
121    
122            for (Classfile classfile : classfiles) {
123                classfile.accept(this);
124            }
125            
126            fireEndSession();
127        }

129        // Classfile
130        public void visitClassfile(Classfile classfile) {
131            Logger.getLogger(getClass()).debug("VisitClassfile():");
132            Logger.getLogger(getClass()).debug("    class = \"" + classfile.getClassName() + "\"");
133    
134            fireBeginClass(classfile);
135            
136            setCurrentMethod(null);
137            setCurrentClass(getMetricsFactory().createClassMetrics(classfile.getClassName()));
138            setCurrentGroup(getCurrentClass().getParent());
139            setCurrentProject(getCurrentGroup().getParent());
140    
141            getMetricsFactory().includeClassMetrics(getCurrentClass());
142    
143            getCurrentProject().addToMeasurement(Metrics.PACKAGES, getCurrentGroup().getName());
144    
145            if ((classfile.getAccessFlag() & Classfile.ACC_PUBLIC) != 0) {
146                getCurrentProject().addToMeasurement(Metrics.PUBLIC_CLASSES);
147                getCurrentGroup().addToMeasurement(Metrics.PUBLIC_CLASSES);
148            }
149    
150            if ((classfile.getAccessFlag() & Classfile.ACC_FINAL) != 0) {
151                getCurrentProject().addToMeasurement(Metrics.FINAL_CLASSES);
152                getCurrentGroup().addToMeasurement(Metrics.FINAL_CLASSES);
153            }
154    
155            if ((classfile.getAccessFlag() & Classfile.ACC_INTERFACE) != 0) {
156                getCurrentProject().addToMeasurement(Metrics.INTERFACES);
157                getCurrentGroup().addToMeasurement(Metrics.INTERFACES);
158            }
159    
160            if ((classfile.getAccessFlag() & Classfile.ACC_ABSTRACT) != 0) {
161                getCurrentProject().addToMeasurement(Metrics.ABSTRACT_CLASSES);
162                getCurrentGroup().addToMeasurement(Metrics.ABSTRACT_CLASSES);
163            }
164    
165            if (classfile.getSuperclassIndex() != 0) {
166                classfile.getRawSuperclass().accept(this);
167    
168                Classfile superclass = classfile.getLoader().getClassfile(classfile.getSuperclassName());
169    
170                if (superclass != null) {
171                    getMetricsFactory().createClassMetrics(superclass.getClassName()).addToMeasurement(Metrics.SUBCLASSES);
172                }
173                getCurrentClass().addToMeasurement(Metrics.DEPTH_OF_INHERITANCE, computeDepthOfInheritance(superclass));
174            }
175    
176            for (Class_info class_info : classfile.getAllInterfaces()) {
177                class_info.accept(this);
178            }
179    
180            for (Field_info field : classfile.getAllFields()) {
181                field.accept(this);
182            }
183    
184            for (Method_info method : classfile.getAllMethods()) {
185                method.accept(this);
186            }
187    
188            sloc = 1;
189            isSynthetic = false;
190    
191            for (Attribute_info attribute : classfile.getAttributes()) {
192                attribute.accept(this);
193            }
194            
195            if (!isSynthetic) {
196                getCurrentClass().addToMeasurement(Metrics.CLASS_SLOC, sloc);
197            }
198    
199            fireEndClass(classfile, getCurrentClass());
200        }

204            Logger.getLogger(getClass()).debug("VisitClass_info():");
205            Logger.getLogger(getClass()).debug("    name = \"" + entry.getName() + "\"");

208            } else {
209                addClassDependency(entry.getName());

242        public void visitField_info(Field_info entry) {
243            getCurrentClass().addToMeasurement(Metrics.ATTRIBUTES);
244    
245            Logger.getLogger(getClass()).debug("VisitField_info(" + entry.getFullSignature() + ")");
246            Logger.getLogger(getClass()).debug("Current class: " + getCurrentClass().getName());
247            Logger.getLogger(getClass()).debug("Access flag: " + entry.getAccessFlag());
248            Logger.getLogger(getClass()).debug("Public: " + (entry.getAccessFlag() & Method_info.ACC_PUBLIC));
249            Logger.getLogger(getClass()).debug("Private: " + (entry.getAccessFlag() & Method_info.ACC_PRIVATE));
250            Logger.getLogger(getClass()).debug("Protected: " + (entry.getAccessFlag() & Method_info.ACC_PROTECTED));
251            Logger.getLogger(getClass()).debug("Static: " + (entry.getAccessFlag() & Method_info.ACC_STATIC));
252            
253            if ((entry.getAccessFlag() & Field_info.ACC_PUBLIC) != 0) {
254                getCurrentClass().addToMeasurement(Metrics.PUBLIC_ATTRIBUTES);
255            } else if ((entry.getAccessFlag() & Field_info.ACC_PRIVATE) != 0) {
256                getCurrentClass().addToMeasurement(Metrics.PRIVATE_ATTRIBUTES);
257            } else if ((entry.getAccessFlag() & Field_info.ACC_PROTECTED) != 0) {
258                getCurrentClass().addToMeasurement(Metrics.PROTECTED_ATTRIBUTES);
259            } else {
260                getCurrentClass().addToMeasurement(Metrics.PACKAGE_ATTRIBUTES);
261            }
262    
263            if ((entry.getAccessFlag() & Field_info.ACC_STATIC) != 0) {
264                getCurrentClass().addToMeasurement(Metrics.STATIC_ATTRIBUTES);
265            }
266    
267            if ((entry.getAccessFlag() & Field_info.ACC_FINAL) != 0) {
268                getCurrentClass().addToMeasurement(Metrics.FINAL_ATTRIBUTES);
269            }
270    
271            if ((entry.getAccessFlag() & Field_info.ACC_VOLATILE) != 0) {
272                getCurrentClass().addToMeasurement(Metrics.VOLATILE_ATTRIBUTES);
273            }
274    
275            if ((entry.getAccessFlag() & Field_info.ACC_TRANSIENT) != 0) {
276                getCurrentClass().addToMeasurement(Metrics.TRANSIENT_ATTRIBUTES);
277            }
278    
279            sloc = 1;
280            isSynthetic = false;
281            
282            super.visitField_info(entry);
283            
284            if (!isSynthetic) {
285                getCurrentClass().addToMeasurement(Metrics.CLASS_SLOC, sloc);
286            }
287    
288            addClassDependencies(processDescriptor(entry.getDescriptor()));
289        }

291        public void visitMethod_info(Method_info entry) {
292            fireBeginMethod(entry);
293            
294            setCurrentMethod(getMetricsFactory().createMethodMetrics(entry.getFullSignature()));
295            getMetricsFactory().includeMethodMetrics(getCurrentMethod());
296            
297            Logger.getLogger(getClass()).debug("VisitMethod_info(" + entry.getFullSignature() + ")");
298            Logger.getLogger(getClass()).debug("Current class: " + getCurrentClass().getName());
299            Logger.getLogger(getClass()).debug("Access flag: " + entry.getAccessFlag());
300            Logger.getLogger(getClass()).debug("Public: " + (entry.getAccessFlag() & Method_info.ACC_PUBLIC));
301            Logger.getLogger(getClass()).debug("Private: " + (entry.getAccessFlag() & Method_info.ACC_PRIVATE));
302            Logger.getLogger(getClass()).debug("Protected: " + (entry.getAccessFlag() & Method_info.ACC_PROTECTED));
303            Logger.getLogger(getClass()).debug("Static: " + (entry.getAccessFlag() & Method_info.ACC_STATIC));
304    
305            sloc = 0;
306            isSynthetic = false;
307            
308            if ((entry.getAccessFlag() & Method_info.ACC_PUBLIC) != 0) {
309                getCurrentClass().addToMeasurement(Metrics.PUBLIC_METHODS);
310            } else if ((entry.getAccessFlag() & Method_info.ACC_PRIVATE) != 0) {
311                getCurrentClass().addToMeasurement(Metrics.PRIVATE_METHODS);
312            } else if ((entry.getAccessFlag() & Method_info.ACC_PROTECTED) != 0) {
313                getCurrentClass().addToMeasurement(Metrics.PROTECTED_METHODS);
314            } else {
315                getCurrentClass().addToMeasurement(Metrics.PACKAGE_METHODS);
316            }
317    
318            if ((entry.getAccessFlag() & Method_info.ACC_STATIC) != 0) {
319                getCurrentClass().addToMeasurement(Metrics.STATIC_METHODS);
320            }
321    
322            if ((entry.getAccessFlag() & Method_info.ACC_FINAL) != 0) {
323                getCurrentClass().addToMeasurement(Metrics.FINAL_METHODS);
324            }
325    
326            if ((entry.getAccessFlag() & Method_info.ACC_SYNCHRONIZED) != 0) {
327                getCurrentClass().addToMeasurement(Metrics.SYNCHRONIZED_METHODS);
328            }
329    
330            if ((entry.getAccessFlag() & Method_info.ACC_NATIVE) != 0) {
331                getCurrentClass().addToMeasurement(Metrics.NATIVE_METHODS);
332            }
333    
334            if ((entry.getAccessFlag() & Method_info.ACC_ABSTRACT) != 0) {
335                getCurrentClass().addToMeasurement(Metrics.ABSTRACT_METHODS);
336                sloc = 1;
337            }
338    
339            getCurrentMethod().addToMeasurement(Metrics.PARAMETERS, SignatureHelper.getParameterCount(entry.getDescriptor()));
340            
341            super.visitMethod_info(entry);
342            
343            if (!isSynthetic) {
344                getCurrentMethod().addToMeasurement(Metrics.SLOC, sloc);
345            }
346    
347            addClassDependencies(processDescriptor(entry.getDescriptor()));
348    
349            fireEndMethod(entry, getCurrentMethod());
350        }

356        public void visitSynthetic_attribute(Synthetic_attribute attribute) {
357            Object owner = attribute.getOwner();
358    
359            isSynthetic = true;
360            
361            if (owner instanceof Classfile) {
362                getCurrentProject().addToMeasurement(Metrics.SYNTHETIC_CLASSES);
363                getCurrentGroup().addToMeasurement(Metrics.SYNTHETIC_CLASSES);
364            } else if (owner instanceof Field_info) {
365                getCurrentClass().addToMeasurement(Metrics.SYNTHETIC_ATTRIBUTES);
366            } else if (owner instanceof Method_info) {
367                getCurrentClass().addToMeasurement(Metrics.SYNTHETIC_METHODS);
368            } else {
369                Logger.getLogger(getClass()).warn("Synthetic attribute on unknown Visitable: " + owner.getClass().getName());
370            }
371        }

392        public void visitInstruction(Instruction helper) {
393            super.visitInstruction(helper);
394    
395            /*
396             *  We can skip the "new" (0xbb) instruction as it is always
397             *  followed by a call to the constructor method.
398             */
399    
400            switch (helper.getOpcode()) {
401                case 0xb2: // getstatic
402                case 0xb3: // putstatic
403                case 0xb4: // getfield
404                case 0xb5: // putfield
405                case 0xb6: // invokevirtual
406                case 0xb7: // invokespecial
407                case 0xb8: // invokestatic
408                case 0xb9: // invokeinterface
409                // case 0xbb: // new
410                case 0xbd: // anewarray
411                case 0xc0: // checkcast
412                case 0xc1: // instanceof
413                case 0xc5: // multianewarray
414                    helper.getIndexedConstantPoolEntry().accept(this);
415                    break;
416                default:
417                    // Do nothing
418                    break;
419            }
420        }

482        private int computeDepthOfInheritance(Classfile classfile) {
483            int result = 1;
484            
485            if (classfile != null && classfile.getSuperclassIndex() != 0) {
486                Classfile superclass = classfile.getLoader().getClassfile(classfile.getSuperclassName());
487                result += computeDepthOfInheritance(superclass);
488            }
489    
490            return result;
491        }

493        private Collection<String> processDescriptor(String str) {
494            Collection<String> result = new LinkedList<String>();
495            
496            Logger.getLogger(getClass()).debug("ProcessDescriptor: " + str);
497    
498            int currentPos = 0;
499            int startPos;
500            int endPos;
501    
502            while ((startPos = str.indexOf('L', currentPos)) != -1) {
503                if ((endPos = str.indexOf(';', startPos)) != -1) {
504                    String classname = SignatureHelper.path2ClassName(str.substring(startPos + 1, endPos));
505                    result.add(classname);
506                    currentPos = endPos + 1;
507                } else {
508                    currentPos = startPos + 1;
509                }
510            }
511    
512            Logger.getLogger(getClass()).debug("ProcessDescriptor: " + result);
513            
514            return result;
515        }

523        private void addClassDependency(String name) {
524            Logger.getLogger(getClass()).debug("AddClassDependency(\"" + name + "\") ...");
525    
526            if (!getCurrentClass().getName().equals(name) && isInFilter(name)) {
527                Metrics other = getMetricsFactory().createClassMetrics(name);
528                    
529                if (getCurrentMethod() != null && isInScope(getCurrentMethod().getName())) {
530                    Logger.getLogger(getClass()).debug("AddClassDependency " + getCurrentMethod().getName() + " -> " + name + " ...");
531                    
532                    if (getCurrentClass().getParent().equals(other.getParent())) {
533                        Logger.getLogger(getClass()).debug("Intra-Package ...");
534                        getCurrentMethod().addToMeasurement(Metrics.OUTBOUND_INTRA_PACKAGE_CLASS_DEPENDENCIES, other.getName());
535                        other.addToMeasurement(Metrics.INBOUND_INTRA_PACKAGE_METHOD_DEPENDENCIES, getCurrentMethod().getName());
536                    } else {
537                        Logger.getLogger(getClass()).debug("Extra-Package ...");
538                        getCurrentMethod().addToMeasurement(Metrics.OUTBOUND_EXTRA_PACKAGE_CLASS_DEPENDENCIES, other.getName());
539                        other.addToMeasurement(Metrics.INBOUND_EXTRA_PACKAGE_METHOD_DEPENDENCIES, getCurrentMethod().getName());
540                    }
541                } else if (isInScope(getCurrentClass().getName())) {
542                    Logger.getLogger(getClass()).debug("AddClassDependency " + getCurrentClass().getName() + " -> " + name + " ...");
543                    
544                    if (getCurrentClass().getParent().equals(other.getParent())) {
545                        Logger.getLogger(getClass()).debug("Intra-Package ...");
546                        getCurrentClass().addToMeasurement(Metrics.OUTBOUND_INTRA_PACKAGE_DEPENDENCIES, other.getName());
547                        other.addToMeasurement(Metrics.INBOUND_INTRA_PACKAGE_DEPENDENCIES, getCurrentClass().getName());
548                    } else {
549                        Logger.getLogger(getClass()).debug("Extra-Package ...");
550                        getCurrentClass().addToMeasurement(Metrics.OUTBOUND_EXTRA_PACKAGE_DEPENDENCIES, other.getName());
551                        other.addToMeasurement(Metrics.INBOUND_EXTRA_PACKAGE_DEPENDENCIES, getCurrentClass().getName());
552                    }
553                }
554            }
555        }
556        
557        private void addMethodDependency(String name) {
558            Logger.getLogger(getClass()).debug("AddMethodDependency " + getCurrentMethod().getName() + " -> " + name + " ...");
559    
560            if (!getCurrentMethod().getName().equals(name) && isInScope(getCurrentMethod().getName()) && isInFilter(name)) {
561                Metrics other = getMetricsFactory().createMethodMetrics(name);
562                
563                if (getCurrentClass().equals(other.getParent())) {
564                    Logger.getLogger(getClass()).debug("Intra-Class ...");
565                    getCurrentMethod().addToMeasurement(Metrics.OUTBOUND_INTRA_CLASS_FEATURE_DEPENDENCIES, other.getName());
566                    other.addToMeasurement(Metrics.INBOUND_INTRA_CLASS_METHOD_DEPENDENCIES, getCurrentMethod().getName());
567                } else if (getCurrentGroup().equals(other.getParent().getParent())) {
568                    Logger.getLogger(getClass()).debug("Intra-Package ...");
569                    getCurrentMethod().addToMeasurement(Metrics.OUTBOUND_INTRA_PACKAGE_FEATURE_DEPENDENCIES, other.getName());
570                    other.addToMeasurement(Metrics.INBOUND_INTRA_PACKAGE_METHOD_DEPENDENCIES, getCurrentMethod().getName());
571                } else {
572                    Logger.getLogger(getClass()).debug("Extra-Package ...");
573                    getCurrentMethod().addToMeasurement(Metrics.OUTBOUND_EXTRA_PACKAGE_FEATURE_DEPENDENCIES, other.getName());
574                    other.addToMeasurement(Metrics.INBOUND_EXTRA_PACKAGE_METHOD_DEPENDENCIES, getCurrentMethod().getName());
575                }
576            }
577        }

579        private boolean isInScope(String name) {
580            boolean result = true;
581    
582            if (scope != null) {
583                result = scope.contains(name);
584            }
585    
586            return result;
587        }
588        
589        private boolean isInFilter(String name) {
590            boolean result = true;
591    
592            if (filter != null) {
