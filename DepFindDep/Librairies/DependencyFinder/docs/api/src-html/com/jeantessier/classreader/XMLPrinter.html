053        public XMLPrinter(PrintWriter out, String encoding, String dtdPrefix) {
054            super(out);
055

059        private void appendHeader(String encoding, String dtdPrefix) {
060            append("<?xml version=\"1.0\" encoding=\"").append(encoding).append("\" ?>").eol();
061            eol();
062            append("<!DOCTYPE classfiles SYSTEM \"").append(dtdPrefix).append("/classfile.dtd\">").eol();
063            eol();
064        }

076        public void visitClassfile(Classfile classfile) {
077            indent().append("<classfile magic-number=\"").append(classfile.getMagicNumber()).append("\" minor-version=\"").append(classfile.getMinorVersion()).append("\" major-version=\"").append(classfile.getMajorVersion()).append("\" access-flag=\"").append(format.format(classfile.getAccessFlag())).append("\">").eol();
078            raiseIndent();
079    
080            top = true;
081            classfile.getConstantPool().accept(this);
082            top = false;
083            
084            if (classfile.isPublic())     indent().append("<public/>").eol();
085            if (classfile.isFinal())      indent().append("<final/>").eol();
086            if (classfile.isSuper())      indent().append("<super/>").eol();
087            if (classfile.isInterface())  indent().append("<is-interface/>").eol();
088            if (classfile.isAbstract())   indent().append("<abstract/>").eol();
089    
090            indent();
091            append("<this-class>");
092            classfile.getRawClass().accept(this);
093            append("</this-class>").eol();
094    
095            indent();
096            append("<superclass>");
097            if (classfile.getSuperclassIndex() != 0) {
098                classfile.getRawSuperclass().accept(this);
099            }
100            append("</superclass>").eol();
101    
102            if (!classfile.getAllInterfaces().isEmpty()) {
103                indent().append("<interfaces>").eol();
104                raiseIndent();
105                for (Class_info class_info : classfile.getAllInterfaces()) {
106                    indent();
107                    append("<interface>");
108                    class_info.accept(this);
109                    append("</interface>").eol();
110                }
111                lowerIndent();
112                indent().append("</interfaces>").eol();
113            }
114            
115            if (!classfile.getAllFields().isEmpty()) {
116                indent().append("<fields>").eol();
117                raiseIndent();
118                for (Field_info field : classfile.getAllFields()) {
119                    field.accept(this);
120                }
121                lowerIndent();
122                indent().append("</fields>").eol();
123            }
124    
125            if (!classfile.getAllMethods().isEmpty()) {
126                indent().append("<methods>").eol();
127                raiseIndent();
128                for (Method_info method : classfile.getAllMethods()) {
129                    method.accept(this);
130                }
131                lowerIndent();
132                indent().append("</methods>").eol();
133            }
134    
135            if (!classfile.getAttributes().isEmpty()) {
136                indent().append("<attributes>").eol();
137                raiseIndent();
138                for (Attribute_info attribute : classfile.getAttributes()) {
139                    attribute.accept(this);
140                }
141                lowerIndent();
142                indent().append("</attributes>").eol();
143            }
144    
145            lowerIndent();
146            indent().append("</classfile>").eol();
147        }

149        public void visitConstantPool(ConstantPool constantPool) {
150            resetCount();
151    
152            indent().append("<constant-pool>").eol();
153            raiseIndent();
154    
155            for (ConstantPoolEntry entry : constantPool) {
156                if (entry != null) {
157                    entry.accept(this);
158                }
159                incrementCount();
160            }
161    
162            lowerIndent();
163            indent().append("</constant-pool>").eol();
164        }

182            Class_info       c   = entry.getRawClass();
183            NameAndType_info nat = entry.getRawNameAndType();
184    
185            if (top) {
186                top = false;
187                indent();
188                append("<field-ref-info id=\"").append(currentCount()).append("\">");
189                append("<class>");
190                c.accept(this);
191                append("</class>");
192                append("<type>");
193                nat.getRawType().accept(this);
194                append("</type>");
195                append("<name>");
196                nat.getRawName().accept(this);
197                append("</name>");
198                append("</field-ref-info>").eol();
199                top = true;
200            } else {
201                append(SignatureHelper.getType(nat.getType()));
202                append(" ");
203                append(entry.getFullSignature());
204            }

388        public void visitMethod_info(Method_info entry) {
389            indent().append("<method-info access-flag=\"").append(format.format(entry.getAccessFlag())).append("\">").eol();
390            raiseIndent();
391    
392            if (entry.isPublic())       indent().append("<public/>").eol();
393            if (entry.isProtected())    indent().append("<protected/>").eol();
394            if (entry.isPrivate())      indent().append("<private/>").eol();
395            if (entry.isStatic())       indent().append("<static/>").eol();
396            if (entry.isFinal())        indent().append("<final/>").eol();
397            if (entry.isSynchronized()) indent().append("<synchronized/>").eol();
398            if (entry.isNative())       indent().append("<native/>").eol();
399            if (entry.isAbstract())     indent().append("<abstract/>").eol();
400            if (entry.isStrict())       indent().append("<strict/>").eol();
401    
402            indent();
403            append("<name>");
404            entry.getRawName().accept(this);
405            append("</name>").eol();
406            
407            if (!entry.getName().equals("<init>") && !entry.getName().equals("<clinit>")) {
408                indent().append("<return-type>").append((entry.getReturnType() != null) ? entry.getReturnType() : "void").append("</return-type>").eol();
409            }
410            indent().append("<signature>").append(entry.getSignature()).append("</signature>").eol();
411    
412            if (!entry.getAttributes().isEmpty()) {
413                indent().append("<attributes>").eol();
414                raiseIndent();
415                super.visitMethod_info(entry);
416                lowerIndent();
417                indent().append("</attributes>").eol();
418            }
419    
420            lowerIndent();
421            indent().append("</method-info>").eol();
422        }

432        public void visitCode_attribute(Code_attribute attribute) {
433            indent().append("<code-attribute>").eol();
434            raiseIndent();
435    
436            indent().append("<length>").append(attribute.getCode().length).append("</length>").eol();
437    
438            indent().append("<instructions>").eol();
439            raiseIndent();
440            Iterator i = attribute.iterator();
441            while (i.hasNext()) {
442                Instruction instr = (Instruction) i.next();
443                indent();
444                append("<instruction pc=\"").append(instr.getStart()).append("\" length=\"").append(instr.getLength()).append("\"");
445                switch (instr.getOpcode()) {
446                    case 0x12: // ldc
447                    case 0x13: // ldc_w
448                    case 0x14: // ldc2_w
449                    case 0xb2: // getstatic
450                    case 0xb3: // putstatic
451                    case 0xb4: // getfield
452                    case 0xb5: // putfield
453                    case 0xb6: // invokevirtual
454                    case 0xb7: // invokespecial
455                    case 0xb8: // invokestatic
456                    case 0xb9: // invokeinterface
457                    case 0xbb: // new
458                    case 0xbd: // anewarray
459                    case 0xc0: // checkcast
460                    case 0xc1: // instanceof
461                    case 0xc5: // multianewarray
462                        append(" index=\"").append(instr.getIndex()).append("\">");
463                        append(instr);
464                        append(" ");
465                        instr.getIndexedConstantPoolEntry().accept(this);
466                        break;
467                    default:
468                        append(">");
469                        append(instr);
470                        break;
471                }
472                append("</instruction>").eol();
473            }
474            lowerIndent();
475            indent().append("</instructions>").eol();
476    
477            if (!attribute.getExceptionHandlers().isEmpty()) {
478                indent().append("<exception-handlers>").eol();
479                raiseIndent();
480                for (ExceptionHandler exceptionHandler : attribute.getExceptionHandlers()) {
481                    exceptionHandler.accept(this);
482                }
483                lowerIndent();
484                indent().append("</exception-handlers>").eol();
485            }
486            
487            if (!attribute.getAttributes().isEmpty()) {
488                indent().append("<attributes>").eol();
489                raiseIndent();
490                for (Attribute_info attribute_info : attribute.getAttributes()) {
491                    attribute_info.accept(this);
492                }
493                lowerIndent();
494                indent().append("</attributes>").eol();
495            }
496            
497            lowerIndent();
498            indent().append("</code-attribute>").eol();
499        }

636        private String escapeXMLCharacters(String text) {
637            String result = text;
638    
639            result = perl.substitute("s/&/&amp;/g", result);
640            result = perl.substitute("s/</&lt;/g", result);
641            result = perl.substitute("s/>/&gt;/g", result);
642            
643            return result;
644        }
