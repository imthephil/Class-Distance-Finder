033    package com.jeantessier.dependencyfinder.cli;
034    
035    import java.io.*;
036    import java.util.*;
037    
038    import com.jeantessier.classreader.*;
039    import com.jeantessier.dependency.*;
040    import com.jeantessier.commandline.*;
041    
042    public class DependencyExtractor extends Command {
043        public DependencyExtractor() throws CommandLineException {
044            super("DependencyExtractor");
045        }
046    
047        protected void showSpecificUsage(PrintStream out) {
048            out.println();
049            out.println("If no files are specified, it processes the current directory.");
050            out.println();
051            out.println("If file is a directory, it is recusively scanned for files");
052            out.println("ending in \".class\".");
053            out.println();
054            out.println("Defaults is text output to the console.");
055            out.println();
056        }
057    
058        protected void populateCommandLineSwitches() {
059            super.populateCommandLineSwitches();
060            populateCommandLineSwitchesForXMLOutput(com.jeantessier.dependency.XMLPrinter.DEFAULT_ENCODING, com.jeantessier.dependency.XMLPrinter.DEFAULT_DTD_PREFIX);
061            populateCommandLineSwitchesForFiltering();
062    
063            getCommandLine().addToggleSwitch("maximize");
064            getCommandLine().addToggleSwitch("minimize");
065    
066            getCommandLine().addToggleSwitch("xml");
067        }
068    
069        protected Collection<CommandLineException> parseCommandLine(String[] args) {
070            Collection<CommandLineException> exceptions = super.parseCommandLine(args);
071    
072            exceptions.addAll(validateCommandLineForFiltering());
073    
074            if (getCommandLine().getToggleSwitch("maximize") && getCommandLine().getToggleSwitch("minimize")) {
075                exceptions.add(new CommandLineException("Only one of -maximize or -minimize is allowed"));
076            }
077    
078            return exceptions;
079        }
080    
081        protected void doProcessing() throws Exception {
082            SelectionCriteria filterCriteria = getFilterCriteria();
083    
084            List<String> parameters = getCommandLine().getParameters();
085            if (parameters.size() == 0) {
086                parameters.add(".");
087            }
088    
089            NodeFactory factory = new NodeFactory();
090            CodeDependencyCollector collector = new CodeDependencyCollector(factory, filterCriteria);
091    
092            ClassfileLoader loader = new TransientClassfileLoader();
093            loader.addLoadListener(new LoadListenerVisitorAdapter(collector));
094            loader.addLoadListener(getVerboseListener());
095            loader.load(parameters);
096    
097            if (getCommandLine().getToggleSwitch("minimize")) {
098                LinkMinimizer minimizer = new LinkMinimizer();
099                minimizer.traverseNodes(factory.getPackages().values());
100            } else if (getCommandLine().getToggleSwitch("maximize")) {
101                LinkMaximizer maximizer = new LinkMaximizer();
102                maximizer.traverseNodes(factory.getPackages().values());
103            }
104    
105            getVerboseListener().print("Printing the graph ...");
106    
107            com.jeantessier.dependency.Printer printer;
108            if (getCommandLine().getToggleSwitch("xml")) {
109                printer = new com.jeantessier.dependency.XMLPrinter(out, getCommandLine().getSingleSwitch("encoding"), getCommandLine().getSingleSwitch("dtd-prefix"));
110            } else {
111                printer = new com.jeantessier.dependency.TextPrinter(out);
112            }
113    
114            if (getCommandLine().isPresent("indent-text")) {
115                printer.setIndentText(getCommandLine().getSingleSwitch("indent-text"));
116            }
117    
118            printer.traverseNodes(factory.getPackages().values());
119        }
120    
121        public static void main(String[] args) throws Exception {
122            new DependencyExtractor().run(args);
123        }
124    }
