033    package com.jeantessier.classreader;
034    
035    import java.io.*;
036    
037    public class NameAndType_info extends ConstantPoolEntry {
038        private int nameIndex;
039        private int typeIndex;
040    
041        public NameAndType_info(ConstantPool constantPool, DataInputStream in) throws IOException {
042            super(constantPool);
043    
044            nameIndex = in.readUnsignedShort();
045            typeIndex = in.readUnsignedShort();
046        }
047    
048        public int getNameIndex() {
049            return nameIndex;
050        }
051    
052        public UTF8_info getRawName() {
053            return (UTF8_info) getConstantPool().get(getNameIndex());
054        }
055    
056        public String getName() {
057            return getRawName().toString();
058        }
059    
060        public int getTypeIndex() {
061            return typeIndex;
062        }
063    
064        public UTF8_info getRawType() {
065            return (UTF8_info) getConstantPool().get(getTypeIndex());
066        }
067    
068        public String getType() {
069            return getRawType().toString();
070        }
071    
072        public String toString() {
073            StringBuffer result = new StringBuffer();
074    
075            result.append(getName()).append(" -> ").append(getType());
076    
077            return result.toString();
078        }
079    
080        public int hashCode() {
081            return getRawName().hashCode() ^ getRawType().hashCode();
082        }
083    
084        public boolean equals(Object object) {
085            boolean result = false;
086    
087            if (this == object) {
088                result = true;
089            } else if (object != null && this.getClass().equals(object.getClass())) {
090                NameAndType_info other = (NameAndType_info) object;
091                result = this.getRawName().equals(other.getRawName()) && this.getRawType().equals(other.getRawType());
092            }
093    
094            return result;
095        }
096    
097        public void accept(Visitor visitor) {
098            visitor.visitNameAndType_info(this);
099        }
100    }
