033    package com.jeantessier.diff;
034    
035    import java.util.*;
036    
037    import com.jeantessier.classreader.*;
038    
039    /**
040     * TODO class comments
041     */
042    public class IncompatibleDifferenceStrategy extends APIDifferenceStrategy {
043        public IncompatibleDifferenceStrategy(DifferenceStrategy strategy) {
044            super(strategy);
045        }
046    
047        public boolean isClassDifferent(Classfile oldClass, Classfile newClass) {
048            return isRemoved(oldClass, newClass) ||
049                   (!isNew(oldClass, newClass) &&
050                    isClassModified(oldClass, newClass));
051        }
052    
053        public boolean isFieldDifferent(Field_info oldField, Field_info newField) {
054            return isRemoved(oldField, newField) ||
055                   isDeprecationModified(oldField, newField) ||
056                   isDeclarationModified(oldField, newField);
057        }
058    
059        public boolean isMethodDifferent(Method_info oldMethod, Method_info newMethod) {
060            return isRemoved(oldMethod, newMethod) ||
061                   isDeprecationModified(oldMethod, newMethod) ||
062                   isDeclarationModified(oldMethod, newMethod);
063        }
064    
065        public boolean isPackageDifferent(Map oldPackage, Map newPackage) {
066            return isPackageRemoved(oldPackage, newPackage) ||
067                   (!isPackageNew(oldPackage, newPackage) &&
068                    isPackageModified(oldPackage, newPackage));
069        }
070    
071        public boolean isDeclarationModified(Classfile oldClass, Classfile newClass) {
072            boolean result = false;
073    
074            result = oldClass != null && newClass != null &&
075                     ((oldClass.isPublic() && !newClass.isPublic()) ||
076                      (oldClass.isInterface() != newClass.isInterface()) ||
077                      (!oldClass.isAbstract() && newClass.isAbstract()) ||
078                      (!oldClass.isFinal() && newClass.isFinal()) ||
079                      isExtendsClauseModified(oldClass, newClass) ||
080                      isImplementsClauseModified(oldClass, newClass));
081    
082            return result;
083        }
084    
085        private boolean isExtendsClauseModified(Classfile oldClass, Classfile newClass) {
086            return !oldClass.getSuperclassName().equals(newClass.getSuperclassName());
087        }
088    
089        private boolean isImplementsClauseModified(Classfile oldClass, Classfile newClass) {
090            return !oldClass.getAllInterfaces().containsAll(newClass.getAllInterfaces()) ||
091                   !newClass.getAllInterfaces().containsAll(oldClass.getAllInterfaces());
092        }
093    
094        private boolean isDeclarationModified(Field_info oldField, Field_info newField) {
095            return oldField != null && newField != null &&
096                   ((oldField.isPublic() && !newField.isPublic()) ||
097                    (oldField.isProtected() && (newField.isPackage() || newField.isPrivate())) ||
098                    (!oldField.isFinal() && newField.isFinal()) ||
099                    !oldField.getType().equals(newField.getType()));
100        }
101    
102        private boolean isDeclarationModified(Method_info oldMethod, Method_info newMethod) {
103            return oldMethod != null && newMethod != null &&
104                   ((oldMethod.isPublic() && !newMethod.isPublic()) ||
105                    (oldMethod.isProtected() && (newMethod.isPackage() || newMethod.isPrivate())) ||
106                    (!oldMethod.isAbstract() && newMethod.isAbstract()) ||
107                    (!oldMethod.isStatic() && newMethod.isStatic()) ||
108                    (!oldMethod.isFinal() && newMethod.isFinal()) ||
109                    !oldMethod.getReturnType().equals(newMethod.getReturnType()) ||
110                    isThrowsClauseModified(oldMethod, newMethod));
111        }
112    
113        private boolean isThrowsClauseModified(Method_info oldMethod, Method_info newMethod) {
114            return !oldMethod.getExceptions().containsAll(newMethod.getExceptions()) ||
115                   !newMethod.getExceptions().containsAll(oldMethod.getExceptions());
116        }
117    
118        protected boolean isDeprecationModified(Deprecatable oldItem, Deprecatable newItem) {
119            return oldItem != null && newItem != null && !oldItem.isDeprecated() && newItem.isDeprecated();
120        }
121    }
