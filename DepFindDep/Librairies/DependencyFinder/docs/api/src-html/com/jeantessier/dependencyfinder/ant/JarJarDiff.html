168        public void execute() throws BuildException {
169            // first off, make sure that we've got what we need
170    
171            if (getOld() == null) {
172                throw new BuildException("old must be set!");
173            }
174    
175            if (getNew() == null) {
176                throw new BuildException("new must be set!");
177            }
178    
179            if (getDestfile() == null) {
180                throw new BuildException("destfile must be set!");
181            }
182    
183            VerboseListener verboseListener = new VerboseListener(this);
184    
185            try {
186                // Collecting data, first classfiles from JARs,
187                // then package/class trees using NodeFactory.
188    
189                log("Loading old classes from path " + getOld());
190                PackageMapper oldPackages = new PackageMapper();
191                ClassfileLoader oldJar = new AggregatingClassfileLoader();
192                oldJar.addLoadListener(oldPackages);
193                oldJar.addLoadListener(verboseListener);
194                oldJar.load(Arrays.asList(getOld().list()));
195    
196                log("Loading new classes from path " + getNew());
197                PackageMapper newPackages = new PackageMapper();
198                ClassfileLoader newJar = new AggregatingClassfileLoader();
199                newJar.addLoadListener(newPackages);
200                newJar.addLoadListener(verboseListener);
201                newJar.load(Arrays.asList(getNew().list()));
202    
203                DifferenceStrategy baseStrategy = getDefaultStrategy(getCode());
204                DifferenceStrategy strategy = getStrategy(getLevel(), baseStrategy);
205    
206                if (getFilter() != null) {
207                    strategy = new ListBasedDifferenceStrategy(strategy, getFilter());
208                }
209    
210                // Starting to compare, first at package level,
211                // then descending to class level for packages
212                // that are in both the old and the new codebase.
213    
214                log("Comparing old and new classes ...");
215    
216                String name = getName();
217                String oldLabel = (getOldlabel() != null) ? getOldlabel() : getOld().toString();
218                String newLabel = (getNewlabel() != null) ? getNewlabel() : getNew().toString();
219    
220                DifferencesFactory factory = new DifferencesFactory(strategy);
221                Differences differences = factory.createProjectDifferences(name, oldLabel, oldPackages, newLabel, newPackages);
222    
223                log("Saving difference report to " + getDestfile().getAbsolutePath());
224    
225                com.jeantessier.diff.Printer printer = new Report(getEncoding(), getDtdprefix());
226                if (getIndenttext() != null) {
227                    printer.setIndentText(getIndenttext());
228                }
229    
230                differences.accept(printer);
231    
232                PrintWriter out = new PrintWriter(new FileWriter(getDestfile()));
233                out.print(printer);
234                out.close();
235            } catch (IOException ex) {
236                throw new BuildException(ex);
237            }
238        }
239    
240        private DifferenceStrategy getStrategy(String level, DifferenceStrategy baseStrategy) {
241            DifferenceStrategy strategy;
242            if (API_STRATEGY.equals(level)) {
243                strategy = new APIDifferenceStrategy(baseStrategy);
244            } else if (INCOMPATIBLE_STRATEGY.equals(level)) {
245                strategy = new IncompatibleDifferenceStrategy(baseStrategy);
246            } else {
247                try {
248                    Constructor constructor;
249                    try {
250                        constructor = Class.forName(level).getConstructor(DifferenceStrategy.class);
251                        strategy = (DifferenceStrategy) constructor.newInstance(baseStrategy);
252                    } catch (NoSuchMethodException ex) {
253                        strategy = (DifferenceStrategy) Class.forName(level).newInstance();
254                    }
255                } catch (InvocationTargetException ex) {
256                    log("Unknown level \"" + level + "\", using default level \"" + DEFAULT_LEVEL + "\": " + ex.getMessage());
257                    strategy = getDefaultStrategy(baseStrategy);
258                } catch (InstantiationException ex) {
259                    log("Unknown level \"" + level + "\", using default level \"" + DEFAULT_LEVEL + "\": " + ex.getMessage());
260                    strategy = getDefaultStrategy(baseStrategy);
261                } catch (IllegalAccessException ex) {
262                    log("Unknown level \"" + level + "\", using default level \"" + DEFAULT_LEVEL + "\": " + ex.getMessage());
263                    strategy = getDefaultStrategy(baseStrategy);
264                } catch (ClassNotFoundException ex) {
265                    log("Unknown level \"" + level + "\", using default level \"" + DEFAULT_LEVEL + "\": " + ex.getMessage());
266                    strategy = getDefaultStrategy(baseStrategy);
267                } catch (ClassCastException ex) {
268                    log("Unknown level \"" + level + "\", using default level \"" + DEFAULT_LEVEL + "\": " + ex.getMessage());
269                    strategy = getDefaultStrategy(baseStrategy);
270                }
271            }
272            return strategy;
273        }
274    
275        private DifferenceStrategy getDefaultStrategy(boolean useCode) {
276            DifferenceStrategy baseStrategy;
277            if (useCode) {
278                baseStrategy = new CodeDifferenceStrategy();
279            } else {
280                baseStrategy = new NoDifferenceStrategy();
281            }
282            return baseStrategy;
283        }
284    
285        private APIDifferenceStrategy getDefaultStrategy(DifferenceStrategy baseStrategy) {
286            return new APIDifferenceStrategy(baseStrategy);
287        }
