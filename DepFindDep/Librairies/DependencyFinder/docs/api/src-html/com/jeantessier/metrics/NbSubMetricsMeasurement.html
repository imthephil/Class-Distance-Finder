033    package com.jeantessier.metrics;
034    
035    import java.io.*;
036    import java.util.*;
037    
038    import org.apache.log4j.*;
039    
040    /**
041     *  Counts the number of submetrics according to selection
042     *  criteria.  If there are no criteria, it matches all
043     *  submetrics.  Each criterion is a boolean expression
044     *  with measurement names, numbers, and boolean operators
045     *  (<, <=, >, >=, ==, and !=).  If a submetric matches
046     *  any one of the expressions in the criteria, it is
047     *  included in the count.
048     *  
049     *  <p>This is the syntax for initializing this type of
050     *  measurement:</p>
051     *  
052     *  <pre>
053     *  &lt;init&gt;
054     *      (number | measurement name [DISPOSE_x]) [operator [(number | measurement name [DISPOSE_x])]]*
055     *      ...
056     *  &lt;/init&gt;
057     *  </pre>
058     */
059    public class NbSubMetricsMeasurement extends MeasurementBase {
060        private static final String OPERATORS = "/(<=)|(<)|(>=)|(>)|(==)|(!=)/";
061    
062        private static final String LESSER_THAN           = "<";
063        private static final String LESSER_THAN_OR_EQUAL  = "<=";
064        private static final String GREATER_THAN          = ">";
065        private static final String GREATER_THAN_OR_EQUAL = ">=";
066        private static final String EQUALS                = "==";
067        private static final String NOT_EQUALS            = "!=";
068    
069        private static final double DELTA = 0.1;
070    
071        private List<String> terms = new LinkedList<String>();
072        private int  value = 0;
073    
074        public NbSubMetricsMeasurement(MeasurementDescriptor descriptor, Metrics context, String initText) {
075            super(descriptor, context, initText);
076    
077            try {
078                BufferedReader in   = new BufferedReader(new StringReader(initText));
079                String         line;
080    
081                while ((line = in.readLine()) != null) {
082                    terms.add(line.trim());
083                }
084    
085                in.close();
086            } catch (Exception ex) {
087                Logger.getLogger(getClass()).debug("Cannot initialize with \"" + initText + "\"", ex);
088                terms.clear();
089            }
090        }
091    
092        public List<String> getTerms() {
093            return terms;
094        }
095        
096        public void accept(MeasurementVisitor visitor) {
097            visitor.visitNbSubMetricsMeasurement(this);
098        }
099    
100        public boolean isEmpty() {
101            if (!isCached()) {
102                compute();
103            }
104    
105            return super.isEmpty();
106        }
107        
108        protected double compute() {
109            if (!isCached()) {
110                synchronized (this) {
111                    if (!isCached()) {
112                        value = 0;
113                        
114                        if (getTerms().isEmpty()) {
115                            value = getContext().getSubMetrics().size();
116                        } else {
117                            for (Metrics metrics : getContext().getSubMetrics()) {
118                                if (getSelectMetrics(metrics)) {
119                                    value++;
120                                }
121                            }
122                        }
123    
124                        setEmpty(value == 0);
125    
126                        setCached(true);
127                    }
128                }
129            }
130            
131            return value;
132        }
133    
134        private boolean getSelectMetrics(Metrics metrics) {
135            boolean result = getTerms().isEmpty();
136            
137            Iterator<String> i = getTerms().iterator();
138            while (!result && i.hasNext()) {
139                result = evaluateTerm(i.next(), metrics);
140            }
141    
142            return result;
143        }
144    
145        private boolean evaluateTerm(String term, Metrics metrics) {
146            boolean result;
147    
148            Logger.getLogger(getClass()).debug("EvaluateTerm(\"" + term + "\", " + metrics + ")");
149            
150            List<String> elements = new ArrayList<String>();
151            perl().split(elements, OPERATORS, term);
152    
153            result = (elements.size() > 0) && ((elements.size() % 2) == 1);
154            
155            if (elements.size() == 1) {
156                result = metrics.hasMeasurement(elements.remove(0));
157            } else {
158                while (result && (elements.size() > 2) && ((elements.size() % 2) == 1)) {
159                    String leftString  = elements.remove(0);
160                    String operator    = elements.remove(0);
161                    String rightString = elements.get(0);
162    
163                    double leftOperand = 0;
164                    try {
165                        leftOperand = Double.parseDouble(leftString);
166                    } catch (NumberFormatException ex) {
167                        try {
168                            leftOperand = resolveOperand(leftString, metrics);
169                        } catch (NullPointerException ex2) {
170                            result = false;
171                        }
172                    }
173    
174                    double rightOperand = 0;
175                    try {
176                        rightOperand = Double.parseDouble(rightString);
177                    } catch (NumberFormatException ex) {
178                        try {
179                            rightOperand = resolveOperand(rightString, metrics);
180                        } catch (NullPointerException ex2) {
181                            result = false;
182                        }
183                    }
184    
185                    if (result) {
186                        if (operator.equals(LESSER_THAN)) {
187                            result = leftOperand < rightOperand;
188                        } else if (operator.equals(LESSER_THAN_OR_EQUAL)) {
189                            result = leftOperand <= rightOperand;
190                        } else if (operator.equals(GREATER_THAN)) {
191                            result = leftOperand > rightOperand;
192                        } else if (operator.equals(GREATER_THAN_OR_EQUAL)) {
193                            result = leftOperand >= rightOperand;
194                        } else if (operator.equals(EQUALS)) {
195                            result = Math.abs(leftOperand - rightOperand) <= DELTA;
196                        } else if (operator.equals(NOT_EQUALS)) {
197                            result = Math.abs(leftOperand - rightOperand) > DELTA;
198                        }
199                    }
200                }
201            }
202    
203            Logger.getLogger(getClass()).debug("EvaluateTerm(\"" + term + "\", " + metrics + "): " + result);
204    
205            return result;
206        }
207    
208        private double resolveOperand(String name, Metrics metrics) {
209            double result = 0;
210                
211            name = name.trim();
212    
213            Logger.getLogger(getClass()).debug("ResolveOperand(\"" + name + "\", " + metrics + ")");
214    
215            if (name.length() != 0) {
216                int dispose;
217    
218                synchronized (perl()) {
219                    if (perl().match("/(.*)\\s+(dispose_\\w+)$/i", name)) {
220                        name = perl().group(1);
221                        
222                        String disposeText = perl().group(2);
223                        
224                        if (disposeText.equalsIgnoreCase("DISPOSE_IGNORE")) {
225                            dispose = StatisticalMeasurement.DISPOSE_IGNORE;
226                        } else if (disposeText.equalsIgnoreCase("DISPOSE_MINIMUM")) {
227                            dispose = StatisticalMeasurement.DISPOSE_MINIMUM;
228                        } else if (disposeText.equalsIgnoreCase("DISPOSE_MEDIAN")) {
229                            dispose = StatisticalMeasurement.DISPOSE_MEDIAN;
230                        } else if (disposeText.equalsIgnoreCase("DISPOSE_AVERAGE")) {
231                            dispose = StatisticalMeasurement.DISPOSE_AVERAGE;
232                        } else if (disposeText.equalsIgnoreCase("DISPOSE_STANDARD_DEVIATION")) {
233                            dispose = StatisticalMeasurement.DISPOSE_STANDARD_DEVIATION;
234                        } else if (disposeText.equalsIgnoreCase("DISPOSE_MAXIMUM")) {
235                            dispose = StatisticalMeasurement.DISPOSE_MAXIMUM;
236                        } else if (disposeText.equalsIgnoreCase("DISPOSE_SUM")) {
237                            dispose = StatisticalMeasurement.DISPOSE_SUM;
238                        } else if (disposeText.equalsIgnoreCase("DISPOSE_NB_DATA_POINTS")) {
239                            dispose = StatisticalMeasurement.DISPOSE_NB_DATA_POINTS;
240                        } else {
241                            dispose = StatisticalMeasurement.DISPOSE_IGNORE;
242                        }
243                    } else {
244                        dispose = StatisticalMeasurement.DISPOSE_IGNORE;
245                    }
246                }
247                
248                Measurement measurement = metrics.getMeasurement(name);
249                
250                if (measurement instanceof StatisticalMeasurement) {
251                    StatisticalMeasurement stats = (StatisticalMeasurement) measurement;
252                    
253                    switch (dispose) {
254                        case StatisticalMeasurement.DISPOSE_MINIMUM:
255                            result = stats.getMinimum();
256                            break;
257                        case StatisticalMeasurement.DISPOSE_MEDIAN:
258                            result = stats.getMedian();
259                            break;
260                        case StatisticalMeasurement.DISPOSE_AVERAGE:
261                            result = stats.getAverage();
262                            break;
263                        case StatisticalMeasurement.DISPOSE_STANDARD_DEVIATION:
264                            result = stats.getStandardDeviation();
265                            break;
266                        case StatisticalMeasurement.DISPOSE_MAXIMUM:
267                            result = stats.getMaximum();
268                            break;
269                        case StatisticalMeasurement.DISPOSE_SUM:
270                            result = stats.getSum();
271                            break;
272                        case StatisticalMeasurement.DISPOSE_NB_DATA_POINTS:
273                            result = stats.getNbDataPoints();
274                            break;
275                        case StatisticalMeasurement.DISPOSE_IGNORE:
276                        default:
277                            result = stats.doubleValue();
278                            break;
279                    }
280                } else if (measurement instanceof NullMeasurement) {
281                    throw new NullPointerException();
282                } else {
283                    result = measurement.doubleValue();
284                }
285            }
286    
287            Logger.getLogger(getClass()).debug("ResolveOperand(\"" + name + "\", " + metrics + "): " + result);
288            
289            return result;
290        }
